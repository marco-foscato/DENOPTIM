158 private ExtractPatternCase getFusedRings() throws Throwable
159 {
160 BiFunction<String, Boolean, Vertex> vertexSupplier =
161 (s, isRCV) -> {
162 int apCount = 0;
163 switch (s) {
164 case "Cl":
165 apCount = 1;
166 break;
167 case "O":
168 apCount = 2;
169 break;
170 case "N":
171 apCount = 3;
172 break;
173 case "C":
174 apCount = 4;
175 break;
176 }
177 return buildFragment(s, apCount, isRCV);
178 };
179
180 /* We label the vertices in order of top left to bottom right. */
181 List<Vertex> vertices = Stream.of(
182 new Pair<>("O", true), new Pair<>("C", false),
183 new Pair<>("Cl", false), new Pair<>("N",false),
184 new Pair<>("C", true), new Pair<>("N", true),
185 new Pair<>("C", true), new Pair<>("O", false),
186 new Pair<>("C", false), new Pair<>("O", true),
187 new Pair<>("C", true), new Pair<>("C", false),
188 new Pair<>("N", true)
189 ).map(p -> vertexSupplier.apply(p.getFirst(), p.getSecond()))
190 .collect(Collectors.toList());
191
192 /* Here we specify the connections between atoms. Previously
193 connected vertices are not connected twice. Chords are not connected. */
194 List<List<Integer>> edges = Arrays.asList(
195 Arrays.asList(1),
196 Arrays.asList(2, 3, 4),
197 Arrays.asList(),
198 Arrays.asList(5, 6),
199 Arrays.asList(),
200 Arrays.asList(),
201 Arrays.asList(7),
202 Arrays.asList(8),
203 Arrays.asList(9, 10),
204 Arrays.asList(),
205 Arrays.asList(11),
206 Arrays.asList(12)
207 );
208
209 DGraph g = null;
210 try
211 {
212 g = buildGraph(vertices, edges);
213 } catch (DENOPTIMException e)
214 {
215 e.printStackTrace();
216 }
217 g.renumberGraphVertices();
218 DGraph.setScaffold(vertices.get(0));
219 addRings(vertices, g);
220 Set<DGraph> expectedSubgraphs = getExpectedSubgraphs(g);
221 return new ExtractPatternCase(g, 2, expectedSubgraphs);
222 }

226 private DGraph buildGraph(List<Vertex> vertices,
227 List<List<Integer>> edges)
228 throws DENOPTIMException {
229 DGraph g = new DGraph();
230 g.addVertex(vertices.get(0));
231 for (int i = 0; i < edges.size(); i++) {
232 Vertex srcVertex = vertices.get(i);
233 for (Integer adj : edges.get(i)) {
234 Vertex trgVertex = vertices.get(adj);
235
236 AttachmentPoint srcAP = srcVertex
237 .getAttachmentPoints()
238 .stream()
239 .filter(ap -> ap.getEdgeUser() == null)
240 .findFirst()
241 .get();
242 AttachmentPoint trgAP = trgVertex
243 .getAttachmentPoints()
244 .stream()
245 .filter(ap -> ap.getEdgeUser() == null)
246 .findFirst()
247 .get();
248
249 try {
250 g.appendVertexOnAP(srcAP, trgAP);
251 } catch (DENOPTIMException e) {
252 // Should not happen
253 e.printStackTrace();
254 }
255 }
256 }
257 return g;
258 }

287 private Set<DGraph> getExpectedSubgraphs(DGraph graph) {
288 List<Set<Integer>> keepVertices = Stream.of(
289 Stream.of(0, 1, 3, 4, 5, 6),
290 Stream.of(8, 9, 10, 11, 12))
291 .map(indices -> indices.collect(Collectors.toSet()))
292 .collect(Collectors.toList());
293
294 List<DGraph> expectedSubgraphs = new ArrayList<>(2);
295 for (Set<Integer> keepVertex : keepVertices) {
296 DGraph expSubgraph = graph.clone();
297 List<Vertex> vertices = expSubgraph.getVertexList();
298 Set<Vertex> removeVertices = IntStream
299 .range(0, vertices.size())
300 .filter(i -> !keepVertex.contains(i))
301 .mapToObj(vertices::get)
302 .collect(Collectors.toSet());
303
304 for (Vertex removeVertex : removeVertices) {
305 expSubgraph.removeVertex(removeVertex);
306 }
307 expectedSubgraphs.add(expSubgraph);
308 }
309
310 return new HashSet<>(expectedSubgraphs);
311 }

359 private DGraph getThreeCycle() throws DENOPTIMException
360 {
361 EmptyVertex v1 = new EmptyVertex(0);
362 EmptyVertex rcv1 = new EmptyVertex(1, new ArrayList<>(),
363 new ArrayList<>(), true);
364 EmptyVertex rcv2 = new EmptyVertex(2, new ArrayList<>(),
365 new ArrayList<>(), true);
366
367 List<EmptyVertex> vertices = Arrays.asList(v1, rcv1, rcv2);
368 for (EmptyVertex v : vertices) {
369 v.setBuildingBlockType(BBType.FRAGMENT);
370 v.addAP();
371 }
372 // Need an additional AP on v1
373 v1.addAP();
374
375 DGraph g = new DGraph();
376 g.addVertex(v1);
377 g.appendVertexOnAP(v1.getAP(0), rcv1.getAP(0));
378 g.appendVertexOnAP(v1.getAP(1), rcv2.getAP(0));
379
380 Ring r = new Ring(new ArrayList<>(
381 Arrays.asList(rcv1, v1, rcv2)));
382 g.addRing(r);
383
384 g.renumberGraphVertices();
385 return g;
386 }

390 private static final class ExtractPatternCase
391 {
392 final DGraph g;
393 final int expectedSize;
394 final Set<DGraph> expectedGraphs;
395 final Comparator<DGraph> graphComparator = (gA, gB) ->
396 gA.sameAs(gB, new StringBuilder()) ? 0 : -1;
397
398 private ExtractPatternCase(DGraph g, int expectedSize,
399 Set<DGraph> expectedGraphs) {
400 this.g = g;
401 this.expectedSize = expectedSize;
402 this.expectedGraphs = expectedGraphs;
403 }
404
405 private boolean matchesExpected(Collection<DGraph> actuals) {
406 if (actuals.size() != expectedSize) {
407 return false;
408 }
409
410 Set<DGraph> unmatchedGraphs = new HashSet<>(expectedGraphs);
411 for (DGraph g : actuals)
412 {
413 boolean hasMatch = expectedGraphs
414 .stream()
415 .anyMatch(exp -> graphComparator.compare(g, exp) == 0);
416 if (hasMatch) {
417 unmatchedGraphs = unmatchedGraphs
418 .stream()
419 .filter(exp -> graphComparator.compare(g, exp) != 0)
420 .collect(Collectors.toSet());
421 } else {
422 return false;
423 }
424 }
425
426 // Check that no graphs are missing from actual
427 return unmatchedGraphs.size() == 0;
428 }
429 }
