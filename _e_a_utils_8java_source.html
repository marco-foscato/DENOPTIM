209 public static int chooseNumberOfSitesToMutate(double[] multiSiteMutationProb,
210 double hit)
211 {
212 double tot = 0;
213 for (int i=0; i<multiSiteMutationProb.length; i++)
214 tot = tot + multiSiteMutationProb[i];
215
216 double scaledHit = hit * tot;
217
218 double min = 0;
219 double max = 0;
220 int choice = 0;
221 for (int i=0; i<multiSiteMutationProb.length; i++)
222 {
223 max = max + multiSiteMutationProb[i];
224 if (min < scaledHit && scaledHit <= max)
225 {
226 choice = i;
227 break;
228 }
229 min = Math.max(min,min+multiSiteMutationProb[i]);
230 }
231 return choice;
232 }

248 public static CandidateSource pickNewCandidateGenerationMode(
249 double xoverWeight, double mutWeight, double newWeight,
250 Randomizer randomizer)
251 {
252 double hit = randomizer.nextDouble()
253 * (xoverWeight + mutWeight + newWeight);
254 if (hit <= xoverWeight)
255 {
256 return CandidateSource.CROSSOVER;
257 } else if (xoverWeight < hit && hit <= (mutWeight+xoverWeight))
258 {
259 return CandidateSource.MUTATION;
260 } else {
261 return CandidateSource.CONSTRUCTION;
262 }
263 }

377 protected static List<Candidate> buildCandidatesByXOver(
378 List<Candidate> eligibleParents, Population population,
379 Monitor mnt, int[] choiceOfParents, int choiceOfXOverSites,
380 int choiceOfOffstring, GAParameters settings,
381 int maxCandidatesToReturn) throws DENOPTIMException
382 {
383 FragmentSpaceParameters fsParams = new FragmentSpaceParameters();
384 if (settings.containsParameters(ParametersType.FS_PARAMS))
385 {
386 fsParams = (FragmentSpaceParameters)settings.getParameters(
387 ParametersType.FS_PARAMS);
388 }
389 FragmentSpace fragSpace = fsParams.getFragmentSpace();
390
391 mnt.increase(CounterID.XOVERATTEMPTS);
392 mnt.increase(CounterID.NEWCANDIDATEATTEMPTS);
393
394 int numatt = 1;
395
396 // Identify a pair of parents that can do crossover, and a pair of
397 // vertexes from which we can define a subgraph (or a branch) to swap
398 XoverSite xos = null;
399 boolean foundPars = false;
400 while (numatt < settings.getMaxGeneticOpAttempts())
401 {
402 if (fragSpace.useAPclassBasedApproach())
403 {
404 xos = EAUtils.performFBCC(eligibleParents,
405 population, choiceOfParents, choiceOfXOverSites,
406 settings);
407 if (xos == null)
408 {
409 numatt++;
410 continue;
411 }
412 } else {
413 //TODO: make it reproducible using choiceOfParents and choiceOfXOverSites
414 Candidate[] parents = EAUtils.selectBasedOnFitness(
415 eligibleParents, 2, settings);
416 if (parents[0] == null || parents[1] == null)
417 {
418 numatt++;
419 continue;
420 }
421 //NB: this does not go into templates!
422 DGraph gpA = parents[0].getGraph();
423 List<Vertex> subGraphA = new ArrayList<Vertex>();
424 gpA.getChildrenTree(EAUtils.selectNonScaffoldNonCapVertex(
425 gpA, settings.getRandomizer()),subGraphA);
426
427 DGraph gpB = parents[1].getGraph();
428 List<Vertex> subGraphB = new ArrayList<Vertex>();
429 gpB.getChildrenTree(EAUtils.selectNonScaffoldNonCapVertex(
430 gpB, settings.getRandomizer()),subGraphB);
431 }
432 foundPars = true;
433 break;
434 }
435 mnt.increaseBy(CounterID.XOVERPARENTSEARCH, numatt);
436
437 if (!foundPars)
438 {
439 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_FINDPARENTS);
440 mnt.increase(CounterID.FAILEDXOVERATTEMPTS);
441 return new ArrayList<Candidate>();
442 }
443
444 Candidate cA = null, cB = null;
445 Vertex vA = null, vB = null;
446 vA = xos.getA().get(0);
447 vB = xos.getB().get(0);
448 DGraph gA = vA.getGraphOwner();
449 cA = gA.getOutermostGraphOwner().getCandidateOwner();
450 DGraph gB = vB.getGraphOwner();
451 cB = gB.getOutermostGraphOwner().getCandidateOwner();
452
453 String candIdA = cA.getName();
454 String candIdB = cB.getName();
455 int gid1 = gA.getGraphId();
456 int gid2 = gB.getGraphId();
457
458 // Start building the offspring
459 XoverSite xosOnClones = xos.projectToClonedGraphs();
460 DGraph gAClone = xosOnClones.getA().get(0).getGraphOwner();
461 DGraph gBClone = xosOnClones.getB().get(0).getGraphOwner();
462
463 try
464 {
465 if (!GraphOperations.performCrossover(xosOnClones,fragSpace))
466 {
467 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_PERFORM);
468 mnt.increase(CounterID.FAILEDXOVERATTEMPTS);
469 return new ArrayList<Candidate>();
470 }
471 } catch (Throwable t) {
472 if (!settings.xoverFailureTolerant)
473 {
474 t.printStackTrace();
475 ArrayList<DGraph> parents = new ArrayList<DGraph>();
476 parents.add(gA);
477 parents.add(gB);
478 DenoptimIO.writeGraphsToSDF(new File(settings.getDataDirectory()
479 + "_failed_xover.sdf"), parents, true,
480 settings.getLogger(), settings.getRandomizer());
481 throw new DENOPTIMException("Error while performing crossover! "+NL
482 + "XOverSite: " + xos.toString() + NL
483 + "XOverSite(C): " + xosOnClones.toString() + NL
484 + " Please, report this to the authors ",t);
485 }
486 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_PERFORM);
487 mnt.increase(CounterID.FAILEDXOVERATTEMPTS);
488 return new ArrayList<Candidate>();
489 }
490 gAClone.setGraphId(GraphUtils.getUniqueGraphIndex());
491 gBClone.setGraphId(GraphUtils.getUniqueGraphIndex());
492 String lstIdVA = "";
493 for (Vertex v : xos.getA())
494 lstIdVA = lstIdVA + "_" + v.getVertexId();
495 String lstIdVB = "";
496 for (Vertex v : xos.getB())
497 lstIdVB = lstIdVB + "_" + v.getVertexId();
498 String[] msgs = new String[2];
499 msgs[0] = "Xover: "
500 + "Gen:" + cA.getGeneration() + " Cand:" + candIdA
501 + "|" + gid1 + "|" + lstIdVA
502 + " X "
503 + "Gen:" + cB.getGeneration() + " Cand:" + candIdB
504 + "|" + gid2 + "|" + lstIdVB;
505 msgs[1] = "Xover: "
506 + "Gen:" + cB.getGeneration() + " Cand:" + candIdB
507 + "|" + gid2 + "|" + lstIdVB
508 + " X "
509 + "Gen:" + cA.getGeneration() + " Cand:" + candIdA
510 + "|" + gid1 + "|" + lstIdVA;
511
512 DGraph[] graphsAffectedByXover = new DGraph[2];
513 graphsAffectedByXover[0] = gAClone;
514 graphsAffectedByXover[1] = gBClone;
515
516 List<Candidate> validOffspring = new Population(settings);
517 for (int ig=0; ig<graphsAffectedByXover.length; ig++)
518 {
519 DGraph g = graphsAffectedByXover[ig];
520
521 // It makes sense to do this on the possibly embedded graph and not
522 // on their embedding owners because there cannot be any new cycle
523 // affecting the latter, but there can be ones affecting the first.
524 if (!EAUtils.setupRings(null, g, settings))
525 {
526 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_SETUPRINGS);
527 continue;
528 }
529
530 // Finalize the graph that is at the outermost level
531 DGraph gOutermost = g.getOutermostGraphOwner();
532 gOutermost.addCappingGroups(fragSpace);
533 gOutermost.renumberGraphVertices();
534 gOutermost.setLocalMsg(msgs[ig]);
535
536 // Consider if the result can be used to define a new candidate
537 Object[] res = null;
538 try
539 {
540 res = gOutermost.checkConsistency(settings);
541 } catch (NullPointerException|IllegalArgumentException e)
542 {
543 if (!settings.xoverGraphFailedEvalTolerant)
544 {
545 ArrayList<DGraph> parents = new ArrayList<DGraph>();
546 parents.add(gA);
547 parents.add(gB);
548 parents.add(gAClone);
549 parents.add(gBClone);
550 DenoptimIO.writeGraphsToSDF(new File(settings.getDataDirectory()
551 + "_failed_xover-ed_check.sdf"), parents, true,
552 settings.getLogger(), settings.getRandomizer());
553 throw e;
554 } else {
555 res = null;
556 }
557 }
558 if (res != null)
559 {
560 if (!EAUtils.setupRings(res, gOutermost, settings))
561 {
562 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_SETUPRINGS);
563 res = null;
564 }
565 } else {
566 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_EVAL);
567 }
568
569 // Check if the chosen combination gives rise to forbidden ends
570 for (Vertex rcv : gOutermost.getFreeRCVertices())
571 {
572 APClass apc = rcv.getEdgeToParent().getSrcAP().getAPClass();
573 if (fragSpace.getCappingMap().get(apc)==null
574 && fragSpace.getForbiddenEndList().contains(apc))
575 {
576 mnt.increase(CounterID.FAILEDXOVERATTEMPTS_FORBENDS);
577 res = null;
578 }
579 }
580 if (res == null)
581 {
582 gOutermost.cleanup();
583 gOutermost = null;
584 continue;
585 }
586
587 // OK: we can now use it to make a new candidate
588 Candidate offspring = new Candidate(gOutermost);
589 offspring.setUID(res[0].toString().trim());
590 offspring.setSmiles(res[1].toString().trim());
591 offspring.setChemicalRepresentation((IAtomContainer) res[2]);
592
593 validOffspring.add(offspring);
594 }
595
596 if (validOffspring.size() == 0)
597 {
598 mnt.increase(CounterID.FAILEDXOVERATTEMPTS);
599 return new ArrayList<Candidate>();
600 }
601
602 if (maxCandidatesToReturn==1)
603 {
604 Candidate chosenOffspring = null;
605 if (choiceOfOffstring<0)
606 {
607 chosenOffspring = settings.getRandomizer().randomlyChooseOne(
608 validOffspring);
609 chosenOffspring.setName("M" + GeneralUtils.getPaddedString(
610 DENOPTIMConstants.MOLDIGITS,
611 GraphUtils.getUniqueMoleculeIndex()));
612 } else {
613 chosenOffspring = validOffspring.get(choiceOfOffstring);
614 }
615 validOffspring.retainAll(Arrays.asList(chosenOffspring));
616 } else {
617 for (Candidate cand : validOffspring)
618 {
619 cand.setName("M" + GeneralUtils.getPaddedString(
620 DENOPTIMConstants.MOLDIGITS,
621 GraphUtils.getUniqueMoleculeIndex()));
622 }
623 }
624 return validOffspring;
625 }

629 protected static Candidate buildCandidateByMutation(
630 List<Candidate> eligibleParents, Monitor mnt,
631 GAParameters settings) throws DENOPTIMException
632 {
633 FragmentSpaceParameters fsParams = new FragmentSpaceParameters();
634 if (settings.containsParameters(ParametersType.FS_PARAMS))
635 {
636 fsParams = (FragmentSpaceParameters)settings.getParameters(
637 ParametersType.FS_PARAMS);
638 }
639 FragmentSpace fragSpace = fsParams.getFragmentSpace();
640
641 mnt.increase(CounterID.MUTATTEMPTS);
642 mnt.increase(CounterID.NEWCANDIDATEATTEMPTS);
643
644 int numatt = 0;
645 Candidate parent = null;
646 while (numatt < settings.getMaxGeneticOpAttempts())
647 {
648 parent = EAUtils.selectBasedOnFitness(eligibleParents,1, settings)[0];
649 if (parent == null)
650 {
651 numatt++;
652 continue;
653 }
654 break;
655 }
656 mnt.increaseBy(CounterID.MUTPARENTSEARCH,numatt);
657 if (parent == null)
658 {
659 mnt.increase(CounterID.FAILEDMUTATTEMTS);
660 return null;
661 }
662
663 DGraph graph = parent.getGraph().clone();
664 graph.renumberGraphVertices();
665
666 String parentMolName = FilenameUtils.getBaseName(parent.getSDFFile());
667 int parentGraphId = parent.getGraph().getGraphId();
668 graph.setLocalMsg("Mutation:"
669 + " Gen:" + parent.getGeneration() + " Cand:" + parentMolName
670 + "|" + parentGraphId);
671
672 if (!GraphOperations.performMutation(graph,mnt,settings))
673 {
674 mnt.increase(CounterID.FAILEDMUTATTEMTS_PERFORM);
675 mnt.increase(CounterID.FAILEDMUTATTEMTS);
676 return null;
677 }
678
679 graph.setGraphId(GraphUtils.getUniqueGraphIndex());
680
681 graph.addCappingGroups(fragSpace);
682
683 Object[] res = null;
684 try
685 {
686 res = graph.checkConsistency(settings);
687 } catch (NullPointerException|IllegalArgumentException e)
688 {
689 if (!settings.mutatedGraphFailedEvalTolerant)
690 {
691 settings.getLogger().log(Level.INFO, "WRITING DEBUG FILE for "
692 + graph.getLocalMsg());
693 DenoptimIO.writeGraphToSDF(new File("debug_evalGrp_parent.sdf"),
694 parent.getGraph(),false, settings.getLogger(),
695 settings.getRandomizer());
696 DenoptimIO.writeGraphToSDF(new File("debug_evalGrp_curr.sdf"),
697 graph,false, settings.getLogger(),
698 settings.getRandomizer());
699 throw e;
700 } else {
701 res = null;
702 mnt.increase(CounterID.FAILEDMUTATTEMTS_EVAL);
703 }
704 }
705
706 if (res != null)
707 {
708 if (!EAUtils.setupRings(res,graph,settings))
709 {
710 res = null;
711 mnt.increase(CounterID.FAILEDMUTATTEMTS_SETUPRINGS);
712 }
713 } else {
714 mnt.increase(CounterID.FAILEDMUTATTEMTS_EVAL);
715 }
716
717 // Check if the chosen combination gives rise to forbidden ends
718 //TODO this should be considered already when making the list of
719 // possible combination of rings
720 for (Vertex rcv : graph.getFreeRCVertices())
721 {
722 APClass apc = rcv.getEdgeToParent().getSrcAP().getAPClass();
723 if (fragSpace.getCappingMap().get(apc)==null
724 && fragSpace.getForbiddenEndList().contains(apc))
725 {
726 res = null;
727 mnt.increase(CounterID.FAILEDMUTATTEMTS_FORBENDS);
728 }
729 }
730
731 if (res == null)
732 {
733 graph.cleanup();
734 graph = null;
735 mnt.increase(CounterID.FAILEDMUTATTEMTS);
736 return null;
737 }
738
739 Candidate offspring = new Candidate(graph);
740 offspring.setUID(res[0].toString().trim());
741 offspring.setSmiles(res[1].toString().trim());
742 offspring.setChemicalRepresentation((IAtomContainer) res[2]);
743 offspring.setName("M" + GeneralUtils.getPaddedString(
744 DENOPTIMConstants.MOLDIGITS,
745 GraphUtils.getUniqueMoleculeIndex()));
746
747 return offspring;
748 }

850 Object[] res = graph.checkConsistency(settings);
851
852 if (res != null)
853 {
854 if (!EAUtils.setupRings(res,graph, settings))
855 {
856 graph.cleanup();
857 mnt.increase(CounterID.FAILEDBUILDATTEMPTS_SETUPRINGS);
858 mnt.increase(CounterID.FAILEDBUILDATTEMPTS);
859 return null;
860 }
861 } else {
862 graph.cleanup();
863 mnt.increase(CounterID.FAILEDBUILDATTEMPTS_EVAL);
864 mnt.increase(CounterID.FAILEDBUILDATTEMPTS);
865 return null;
866 }
867
868 // Check if the chosen combination gives rise to forbidden ends
869 //TODO: this should be considered already when making the list of
870 // possible combination of rings
871 for (Vertex rcv : graph.getFreeRCVertices())
872 {
873 // Also exclude any RCV that is not bound to anything?
874 if (rcv.getEdgeToParent() == null)
875 {
876 res = null;
877 mnt.increase(CounterID.FAILEDBUILDATTEMPTS_FORBIDENDS);
878 }
879 if (rcv.getEdgeToParent() == null)
880 {
881 // RCV as scaffold! Ignore special case
882 continue;
883 }
884 APClass apc = rcv.getEdgeToParent().getSrcAP().getAPClass();

886 && fragSpace.getForbiddenEndList().contains(apc))

888 res = null;
889 mnt.increase(CounterID.FAILEDBUILDATTEMPTS_FORBIDENDS);
890 }
891 }

895 graph.cleanup();
896 mnt.increase(CounterID.FAILEDBUILDATTEMPTS);
897 return null;
898 }
899
900 Candidate candidate = new Candidate(graph);
901 candidate.setUID(res[0].toString().trim());
902 candidate.setSmiles(res[1].toString().trim());
903 candidate.setChemicalRepresentation((IAtomContainer) res[2]);
904
905 candidate.setName("M" + GeneralUtils.getPaddedString(
906 DENOPTIMConstants.MOLDIGITS,
907 GraphUtils.getUniqueMoleculeIndex()));

1052 public static DGraph makeGraphFromFragmentationOfMol(IAtomContainer mol,
1053 List<CuttingRule> cuttingRules, Logger logger,
1054 ScaffoldingPolicy scaffoldingPolicy, double linearAngleLimit)
1055 throws DENOPTIMException
1056 {
1057 // We expect only Fragments here.
1058 List<Vertex> fragments = FragmenterTools.fragmentation(mol,
1059 cuttingRules, logger);
1060 for (Vertex v : fragments)
1061 {
1062 Fragment frag = (Fragment) v;
1063
1064 // This is done to set the symmetry relations in each vertex
1065 frag.updateAPs();
1066
1067 // Add linearity-breaking dummy atoms
1068 DummyAtomHandler.addDummiesOnLinearities(frag, linearAngleLimit);
1069 }
1070 if (fragments.size()==0)
1071 {
1072 throw new DENOPTIMException("Fragmentation of molecule with "
1073 + mol.getAtomCount() + " atoms produced 0 fragments.");
1074 }
1075
1076 // Define which fragment is the scaffold
1077 Vertex scaffold = null;
1078 switch (scaffoldingPolicy)
1079 {
1080 case ELEMENT:
1081 {
1082 for (Vertex v : fragments)
1083 {
1084 if (v instanceof Fragment)
1085 {
1086 boolean setAsScaffold = false;
1087 IAtomContainer iac = v.getIAtomContainer();
1088 for (IAtom atm : iac.atoms())
1089 {
1090 if (scaffoldingPolicy.label.equals(
1091 MoleculeUtils.getSymbolOrLabel(atm)))
1092 {
1093 setAsScaffold = true;
1094 break;
1095 }
1096 }
1097 if (setAsScaffold)
1098 {
1099 scaffold = v;
1100 break;
1101 }
1102 }
1103 }
1104 break;
1105 }
1106
1107 default:
1108 case LARGEST_FRAGMENT:
1109 {
1110 try {
1111 scaffold = fragments.stream()
1112 .max(Comparator.comparing(
1113 Vertex::getHeavyAtomsCount))
1114 .get();
1115 } catch (Exception e)
1116 {
1117 throw new DENOPTIMException("Cannot get largest fragment "
1118 + "among " + fragments.size() + " fragments.", e);
1119 }
1120 break;
1121 }
1122 }
1123 if (scaffold==null)
1124 {
1125 throw new DENOPTIMException("No fragment matches criteria to be "
1126 + "identified as the "
1127 + BBType.SCAFFOLD.toString().toLowerCase() + ".");
1128 }
1129 scaffold.setVertexId(0);
1130 scaffold.setBuildingBlockType(BBType.SCAFFOLD);
1131
1132 // Build the graph
1133 DGraph graph = new DGraph();
1134 graph.addVertex(scaffold);
1135 AtomicInteger vId = new AtomicInteger(1);
1136 for (int i=1; i<fragments.size(); i++)
1137 {
1138 appendVertexesToGraphFollowingEdges(graph, vId, fragments);
1139 }
1140
1141 // Set symmetry relations: these depend on which scaffold we have chosen
1142 graph.detectSymVertexSets();
1143
1144 return graph;
1145 }

1148
1149 private static void appendVertexesToGraphFollowingEdges(DGraph graph,
1150 AtomicInteger vId, List<Vertex> vertexes) throws DENOPTIMException
1151 {
1152 // We seek for the last and non-RCV vertex added to the graph
1153 Vertex lastlyAdded = null;
1154 for (int i=-1; i>-4; i--)
1155 {
1156 lastlyAdded = graph.getVertexList().get(
1157 graph.getVertexList().size()+i);
1158 if (!lastlyAdded.isRCV())
1159 break;
1160 }
1161 for (AttachmentPoint apI : lastlyAdded.getAttachmentPoints())
1162 {
1163 if (!apI.isAvailable())
1164 continue;
1165
1166 for (int j=0; j<vertexes.size(); j++)
1167 {
1168 Vertex fragJ = vertexes.get(j);
1169
1170 boolean ringClosure = false;
1171 if (graph.containsVertex(fragJ))
1172 {
1173 ringClosure = true;
1174 }
1175 for (AttachmentPoint apJ : fragJ.getAttachmentPoints())
1176 {
1177 if (apI==apJ)
1178 continue;
1179
1180 if (apI.getCutId()==apJ.getCutId())
1181 {
1182 if (ringClosure)
1183 {
1184 Vertex rcvI = FragmenterTools.getRCPForAP(apI,
1185 APClass.make(APClass.ATPLUS, 0,
1186 BondType.ANY));
1187 rcvI.setBuildingBlockType(BBType.FRAGMENT);
1188 rcvI.setVertexId(vId.getAndIncrement());
1189 graph.appendVertexOnAP(apI, rcvI.getAP(0));
1190
1191 Vertex rcvJ = FragmenterTools.getRCPForAP(apJ,
1192 APClass.make(APClass.ATMINUS, 0,
1193 BondType.ANY));
1194 rcvJ.setBuildingBlockType(BBType.FRAGMENT);
1195 rcvJ.setVertexId(vId.getAndIncrement());
1196 graph.appendVertexOnAP(apJ, rcvJ.getAP(0));
1197 graph.addRing(rcvI, rcvJ);
1198 } else {
1199 fragJ.setBuildingBlockType(BBType.FRAGMENT);
1200 fragJ.setVertexId(vId.getAndIncrement());
1201 graph.appendVertexOnAP(apI, apJ);
1202
1203 // Recursion into the branch of the graph that is
1204 // rooted onto the lastly added vertex
1205 appendVertexesToGraphFollowingEdges(graph, vId,
1206 vertexes);
1207 }
1208 }
1209 }
1210 }
1211 }
