526 public Vertex getVertexFromLibrary(
527 Vertex.BBType bbType, int bbIdx) throws DENOPTIMException
528 {
529 String msg = "";
530 switch (bbType)
531 {
532 case SCAFFOLD:
533 if (scaffoldLib == null)
534 {
535 msg = "Cannot retrieve scaffolds before initialising the "
536 + "scaffold library.";
537 throw new DENOPTIMException(msg);
538 }
539 break;
540 case FRAGMENT:
541 if (fragmentLib == null)
542 {
543 msg = "Cannot retrieve fragments before initialising the "
544 + "fragment library.";
545 throw new DENOPTIMException(msg);
546 }
547 break;
548 case CAP:
549 if (cappingLib == null)
550 {
551 msg = "Cannot retrieve capping groups before initialising"
552 + "the library of capping groups.";
553 throw new DENOPTIMException(msg);
554 }
555 break;
556
557 default:
558 throw new DENOPTIMException("Cannot find building block of "
559 + "type '" + bbType + "' in the fragment space.");
560 }
561
562 Vertex originalVrtx = null;
563 switch (bbType)
564 {
565 case SCAFFOLD:
566 if (bbIdx >-1 && bbIdx < scaffoldLib.size())
567 {
568 originalVrtx = scaffoldLib.get(bbIdx);
569 }
570 else
571 {
572 msg = "Mismatch between scaffold bbIdx (" + bbIdx
573 + ") and size of the library (" + scaffoldLib.size()
574 + "). FragType: " + bbType;
575 settings.getLogger().log(Level.SEVERE, msg);
576 throw new DENOPTIMException(msg);
577 }
578 break;
579
580 case FRAGMENT:
581 if (bbIdx >-1 && bbIdx < fragmentLib.size())
582 {
583 originalVrtx = fragmentLib.get(bbIdx);
584 }
585 else
586 {
587 msg = "Mismatch between fragment bbIdx (" + bbIdx
588 + ") and size of the library (" + fragmentLib.size()
589 + "). FragType: " + bbType;
590 settings.getLogger().log(Level.SEVERE, msg);
591 throw new DENOPTIMException(msg);
592 }
593 break;
594
595 case CAP:
596 if (bbIdx >-1 && bbIdx < cappingLib.size())
597 {
598 originalVrtx = cappingLib.get(bbIdx);
599 }
600 else
601 {
602 msg = "Mismatch between capping group bbIdx " + bbIdx
603 + ") and size of the library (" + cappingLib.size()
604 + "). FragType: " + bbType;
605 settings.getLogger().log(Level.SEVERE, msg);
606 throw new DENOPTIMException(msg);
607 }
608 break;
609
610 case UNDEFINED:
611 msg = "Attempting to take UNDEFINED type of building block from "
612 + "fragment library.";
613 settings.getLogger().log(Level.WARNING, msg);
614 if (bbIdx < fragmentLib.size())
615 {
616 originalVrtx = fragmentLib.get(bbIdx);
617 }
618 else
619 {
620 msg = "Mismatch between fragment bbIdx (" + bbIdx
621 + ") and size of the library (" + fragmentLib.size()
622 + "). FragType: " + bbType;
623 settings.getLogger().log(Level.SEVERE, msg);
624 throw new DENOPTIMException(msg);
625 }
626 break;
627
628 default:
629 msg = "Unknown type of fragment '" + bbType + "'.";
630 settings.getLogger().log(Level.SEVERE, msg);
631 throw new DENOPTIMException(msg);
632 }
633 Vertex clone = originalVrtx.clone();
634
635 clone.setVertexId(GraphUtils.getUniqueVertexIndex());
636
637 clone.setBuildingBlockId(bbIdx);
638 if (originalVrtx.getBuildingBlockId() != bbIdx)
639 {
640 settings.getLogger().log(Level.WARNING, "Mismatch between building "
641 + "block ID ("
642 + originalVrtx.getBuildingBlockId() + ") and position in "
643 + "the list of building blocks (" + bbIdx + ") for type "
644 + bbType + ".");
645 }
646 return clone;
647 }

1060 ArrayList<AttachmentPoint> srcAPs)

1063 new ArrayList<AttachmentPoint>();

1065 for (AttachmentPoint ap : srcAPs)

1069
1070 if (first)
1071 {

1073 first = false;
1074 continue;
1075 }
1076
1077 ArrayList<AttachmentPoint> toKeep =
1078 new ArrayList<AttachmentPoint>();
1079 for (AttachmentPoint candAp : compAps)
1080 {
1081 for (AttachmentPoint newCand : compForOne)

1083 if (newCand == candAp)
1084 {
1085 toKeep.add(candAp);
1086 break;
1087 }
1088 }
1089 }
1090
1091 compAps = toKeep;
1092

1095 break;
1096 }
1097 }
