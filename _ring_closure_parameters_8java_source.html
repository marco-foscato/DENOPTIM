447 public void interpretKeyword(String key, String value)
448 throws DENOPTIMException
449 {
450 String msg = "";
451 switch (key.toUpperCase())
452 {
453 case "VERBOSITY=":
454 try
455 {
456 verbosity = Integer.parseInt(value);
457 }
458 catch (Throwable t)
459 {
460 msg = "Unable to understand value '" + value + "'";
461 throw new DENOPTIMException(msg);
462 }
463 break;
464 case "CLOSERINGS":
465 closeRings = true;
466 break;
467 case "CLOSERINGS=":
468 closeRings = true;
469 break;
470 case "BUILDCHELATESMODE":
471 buildChelatesMode = true;
472 break;
473 case "ORPHANAPCLASS=":
474 buildChelatesMode = true;
475 metalCoordinatingAPClasses.add(APClass.make(value));
476 break;
477 case "SELECTFRAGMENTSFROMCLOSABLECHAINS":
478 selectFragsFromCC = true;
479 break;
480 case "EVALUATIONCLOSABILITYMODE=":
481 switch (value.toUpperCase())
482 {
483 case "RING_SIZE":
484 rceMode = -1;
485 break;
486
487 case "CONSTITUTION":
488 rceMode = 0;
489 break;
490
491 case "3D-CONFORMATION":
492 rceMode = 1;
493 break;
494
495 case "CONSTITUTION_AND_3D-CONFORMATION":
496 rceMode = 2;
497 break;
498
499 default:
500 msg = "Unable to understand value '" + value + "'";
501 throw new DENOPTIMException(msg);
502 }
503 break;
504 case "MAXROTBONDS=":
505 try
506 {
507 maxRotBonds = Integer.parseInt(value);
508 }
509 catch (Throwable t)
510 {
511 msg = "Unable to understand value '" + value + "'";
512 throw new DENOPTIMException(msg);
513 }
514 break;
515 case "MAXSIZENEWRINGS=":
516 try
517 {
518 maxRingSize = Integer.parseInt(value);
519 }
520 catch (Throwable t)
521 {
522 msg = "Unable to understand value '" + value + "'";
523 throw new DENOPTIMException(msg);
524 }
525 break;
526 case "MINRCAPERTYPEPERGRAPH=":
527 try
528 {
529 String[] words = value.trim().split("\\s+");
530 if (words.length!=2)
531 {
532 msg = "Unable to understand value '" + value + "'";
533 throw new DENOPTIMException(msg);
534 }
535 minRcaPerType.put(words[1],Integer.parseInt(words[0]));
536 }
537 catch (Throwable t)
538 {
539 msg = "Unable to understand value '" + value + "'";
540 throw new DENOPTIMException(msg);
541 }
542 break;
543 case "MAXRCAPERTYPEPERGRAPH=":
544 try
545 {
546 String[] words = value.trim().split("\\s+");
547 if (words.length!=2)
548 {
549 msg = "Unable to understand value '" + value + "'";
550 throw new DENOPTIMException(msg);
551 }
552 maxRcaPerType.put(words[1],Integer.parseInt(words[0]));
553 }
554 catch (Throwable t)
555 {
556 msg = "Unable to understand value '" + value + "'";
557 throw new DENOPTIMException(msg);
558 }
559 break;
560 case "MINNUMBEROFRINGCLOSURES=":
561 try
562 {
563 minRingClosures = Integer.parseInt(value);
564 }
565 catch (Throwable t)
566 {
567 msg = "Unable to understand value '" + value + "'";
568 throw new DENOPTIMException(msg);
569 }
570 break;
571 case "MAXNUMBERRINGCLOSURES=":
572 try
573 {
574 maxRingClosures = Integer.parseInt(value);
575 }
576 catch (Throwable t)
577 {
578 msg = "Unable to understand value '" + value + "'";
579 throw new DENOPTIMException(msg);
580 }
581 break;
582 case "LINEARITYLIMIT=":
583 try
584 {
585 linearityLimit = Double.parseDouble(value);
586 }
587 catch (Throwable t)
588 {
589 msg = "Unable to understand value '" + value + "'";
590 throw new DENOPTIMException(msg);
591 }
592 break;
593 case "DISTANCETOLERANCEFACTOR=":
594 try
595 {
596 rcTolDist = Double.parseDouble(value);
597 }
598 catch (Throwable t)
599 {
600 msg = "Unable to understand value '" + value + "'";
601 throw new DENOPTIMException(msg);
602 }
603 break;
604 case "EXTRADISTANCETOLERANCEFACTOR=":
605 try
606 {
607 pathConfSearchExtraTol = Double.parseDouble(value);
608 }
609 catch (Throwable t)
610 {
611 msg = "Unable to understand value '" + value + "'";
612 throw new DENOPTIMException(msg);
613 }
614 break;
615 case "MAXDOTPROD=":
616 try
617 {
618 rcMaxDot = Double.parseDouble(value);
619 }
620 catch (Throwable t)
621 {
622 msg = "Unable to understand value '" + value + "'";
623 throw new DENOPTIMException(msg);
624 }
625 break;
626 case "CONFSEARCHSTEP=":
627 try
628 {
629 pathConfSearchStep = Double.parseDouble(value);
630 }
631 catch (Throwable t)
632 {
633 msg = "Unable to understand value '" + value + "'";
634 throw new DENOPTIMException(msg);
635 }
636 break;
637 case "RINGSIZEBIAS=":
638 String[] words = value.split("\\s+");
639 if (words.length != 2)
640 {
641 msg = "Unable to read ring size and bias wevight from "
642 + "'" + value + "'. Expected syntax is: "
643 + "'5 2' as to specify that 5-member rings "
644 + "are given a weight of 2.";
645 throw new DENOPTIMException(msg);
646 }
647 int size = -1;
648 int weight = 0;
649 try
650 {
651 size = Integer.parseInt(words[0]);
652 } catch (Throwable t)
653 {
654 throw new DENOPTIMException("Ring size must be an integer");
655 }
656 try
657 {
658 weight = Integer.parseInt(words[1]);
659 } catch (Throwable t)
660 {
661 throw new DENOPTIMException("Bias weight must be an integer");
662 }
663 if (size < 0)
664 {
665 msg = "Ring size must be >= 0";
666 throw new DENOPTIMException(msg);
667 }
668 if (weight < 0)
669 {
670 msg = "Bias for ring size must be >= 0";
671 throw new DENOPTIMException(msg);
672 }
673 if (size > maxRingSize)
674 {
675 for (int i=maxRingSize+1; i<size; i++)
676 {
677 ringSizeBias.add(0);
678 }
679 ringSizeBias.add(weight);
680 maxRingSize = size;
681 }
682 else
683 {
684 ringSizeBias.set(size,weight);
685 }
686 break;
687 case "CLOSABLERINGSMARTS=":
688 int id = ringClosabCondAsSMARTS.size();
689 ringClosabCondAsSMARTS.put("q"+id,value);
690 break;
691 case "REQUIREDELEMENTINRINGS=":
692 reqElInRings.add(value);
693 break;
694 case "RCCINDEX=":
695 rccIndex = value;
696 break;
697 case "RCCFOLDER=":
698 serializeRCCs = true;
699 rccFolder = value;
700 break;
701 case "EXHAUSTIVECONFSEARCH":
702 exhaustiveConfSrch = true;
703 break;
704 case "CHECKINTERDEPENDENTCHAINS":
705 checkInterdepPaths = true;
706 break;
707 default:
708 msg = "Keyword '" + key + "' is not a known ring closure-"
709 + "related keyword. Check input files.";
710 throw new DENOPTIMException(msg);
711 }
712 }

716 public void checkParameters() throws DENOPTIMException
717 {
718 String msg = "";
719 if (!closeRings)
720 {
721 msg = "The use of ring-closure related keywords "
722 + "is dependent on the activation of the ring-closing "
723 + "machinery (use "
724 + ParametersType.RC_PARAMS.getKeywordRoot()
725 + "CLOSERINGS keyword).";
726 throw new DENOPTIMException(msg);
727 }
728
729 if (rceMode > 2)
730 {
731 msg = "Unknown ring-closure evaluation mode '" + rceMode + "' "
732 + "Acceptable values are: "
733 + "-1 (only ring size)"
734 + "0 (only contitutional), "
735 + "1 (only 3D chain), "
736 + "2 (both constitutional and 3D chain). ";
737 throw new DENOPTIMException(msg);
738 }
739
740 if (rccFolder!="" && !FileUtils.checkExists(rccFolder))
741 {
742 File folder = new File(rccFolder);
743 try
744 {
745 folder.mkdir();
746 }
747 catch (Throwable t2)
748 {
749 msg = "CyclicGraphHandler can't make folder " + rccFolder
750 + ". " + t2;
751 throw new DENOPTIMException(msg);
752 }
753 }
754
755 if (buildChelatesMode && !closeRings)
756 {
757 msg = "To use the BuildChelates mode you need to perform "
758 + "experiments capable of closing/opening and modifying "
759 + "rings. See the "
760 + ParametersType.RC_PARAMS.getKeywordRoot()
761 + "CLOSERINGS keyword "
762 + "and provide the related input (i.e., ring "
763 + "closing attractor fragments).";
764 throw new DENOPTIMException(msg);
765 }
766
767 for (String type : minRcaPerType.keySet())
768 {
769 if (minRcaPerType.get(type) > maxRcaPerType.get(type))
770 {
771 msg = "Check values of max and min number of RCA for type '"
772 + type+"'.";
773 throw new DENOPTIMException(msg);
774 }
775 }
776
777 if (minRingClosures > maxRingClosures)
778 {
779 msg = "Check values of minRingClosures and maxRingClosures";
780 throw new DENOPTIMException(msg);
781 }
782
783 if (checkInterdepPaths && !exhaustiveConfSrch)
784 {
785 exhaustiveConfSrch = true;
786 }
787
788 if (exhaustiveConfSrch)
789 {
790 msg = "Exhaustive conformational search has been turned ON. "
791 + "This is a very time consuming task!";
792 StaticLogger.appLogger.log(Level.WARNING,msg);
793 }
794
795 if (selectFragsFromCC)
796 {
797 msg = "DENOPTIM can guide the selection of fragments to "
798 + "those leading to known closable chains. This "
799 + "functionality is currently under development "
800 + "and is fully operative only for rings "
801 + "involving the scaffolds. ";
802 StaticLogger.appLogger.log(Level.WARNING,msg);
803 }
804 checkOtherParameters();
805 }
