193 public RingClosingAttractor(IAtom atm, IAtomContainer mol)
194 {
195 this.atm = atm;
196 for (String atyp : DENOPTIMConstants.RCATYPEMAP.keySet())
197 {
198 if (MoleculeUtils.getSymbolOrLabel(atm).equals(atyp))
199 {
200 this.attType = atyp;
201 this.attA11 = paramAR11.get(atyp);
202 this.attA12 = paramAR12.get(atyp);
203 this.attB11 = paramBR11.get(atyp);
204 this.attB12 = paramBR12.get(atyp);
205 break;
206 }
207 }
208 // We can try to build RCAs from any atom, but only those atoms that
209 // really are RCAs do have the information we are extracting here
210 if (isAttractor())
211 {
212 //Well, any atom might have neighbors, but RCA only have 1. So
213 // the assumption is valid for RCAs.
214 this.src = mol.getConnectedAtomsList(atm).get(0);
215 // But, there are properties that can be found only in relation to
216 // actual RCAs (whether used to make a ring, or not)
217 this.apClass = (APClass) atm.getProperty(
218 DENOPTIMConstants.RCAPROPAPCTORCA);
219 this.bndTyp = (BondType) atm.getProperty(
220 DENOPTIMConstants.RCAPROPCHORDBNDTYP);
221 this.ringUser = (Ring) atm.getProperty(
222 DENOPTIMConstants.RCAPROPRINGUSER);
223 }
224 }
