832 throws DENOPTIMException
833 {
834 String msg = "";
835 switch (key.toUpperCase())
836 {
837 case "VERBOSITY=":
838 {
839 verbosity = Integer.parseInt(value);
840 break;
841 }
842
843 case "NUMPARALLELTASKS=":
844 {
845 if (value.length() > 0)
846 {
847 numParallelTasks = Integer.parseInt(value);
848 }
849 break;
850 }
851
852 case "PARALLELIZATION=":
853 {
854 switch (value.toUpperCase())
855 {
856 case "SYNCHRONOUS":
857 parallelizationScheme = 1;
858 break;
859 case "ASYNCHRONOUS":
860 parallelizationScheme = 2;
861 break;
862 default:
863 throw new DENOPTIMException("Unknown parallelization scheme.");
864 }
865 break;
866 }
867
868 case "PRECISIONLEVEL=":
869 {
870 if (value.length() > 0)
871 {
872 precisionLevel = Integer.parseInt(value);
873 }
874 break;
875 }
876
877 case "UIDFILEIN=":
878 {
879 if (value.length() > 0)
880 {
881 uidFileIn = value;
882 }
883 break;
884 }
885
886 case "UIDFILEOUT=":
887 {
888 if (value.length() > 0)
889 {
890 uidFileOut = value;
891 }
892 break;
893 }
894
895 case "MONITORFILE=":
896 {
897 if (value.length() > 0)
898 {
899 monitorFile = value;
900 }
901 break;
902 }
903
904 case "MONITORDUMPSTEP=":
905 {
906 if (value.length() > 0)
907 {
908 monitorDumpStep = Integer.parseInt(value);
909 dumpMonitor = true;
910 }
911 break;
912 }
913
914 case "RANDOMSEED=":

916 if (value.length() > 0)
917 {
918 seed = Long.parseLong(value);
919 }
920 break;
921 }

923 case "MAXTRIESPERPOPULATION=":

926 maxTriesPerPop = Integer.parseInt(value);

928 }

1072 }

1075 {
1076 crowdingProbabilityScheme = convertProbabilityScheme(value);
1077 break;
1078 }
1079
1080 case "NUMGENERATIONS=":
1081 {
1082 if (value.length() > 0)
1083 {
1084 numGenerations = Integer.parseInt(value);
1085 }
1086 break;
1087 }
1088
1089 case "NUMCHILDREN=":
1090 {
1091 if (value.length() > 0)
1092 {
1093 numOfChildren = Integer.parseInt(value);
1094 }
1095 break;
1096 }
1097
1098 case "CROSSOVERWEIGHT=":

1102 crossoverWeight = Double.parseDouble(value);

1104 break;
1105 }
1106
1107 case "MUTATIONWEIGHT=":
1108 {
1109 if (value.length() > 0)
1110 {
1111 mutationWeight = Double.parseDouble(value);
1112 }
1113 break;
1114 }
1115
1116 case "COUPLEMUTATIONTOCROSSOVER=":
1117 {
1118 if (value.length() > 0)
1119 {
1120 coupleMutationAndCrossover = readYesNoTrueFalse(value);
1121 }
1122 break;
1123 }
1124
1125 case "PARENTSSURVIVE=":
1126 {
1127 if (value.length() > 0)
1128 {
1129 parentsSurvive = readYesNoTrueFalse(value);
1130 }
1131 break;
1132 }
1133
1134 case "EXCLUDEMUTATIONTYPE=":
1135 {
1136 if (value.length() > 0)
1137 {
1138 excludedMutationTypes.add(MutationType.valueOf(value));
1139 }
1140 break;
1141 }
1142
1143 case "CONSTRUCTIONWEIGHT=":
1144 {
1145 if (value.length() > 0)
1146 {
1147 builtAnewWeight = Double.parseDouble(value);
1148 }
1149 break;
1150 }
1151
1152 case "REPLACEMENTSTRATEGY=":
1153 {
1154 switch (value.toUpperCase())
1155 {
1156 case "NONE":
1157 replacementStrategy = 2;
1158 break;
1159 case "ELITIST":
1160 replacementStrategy = 1;
1161 break;
1162 default:
1163 throw new DENOPTIMException("Unknown replacement strategy.");
1164 }
1165 break;
1166 }
1167
1168 case "POPULATIONSIZE=":
1169 {
1170 if (value.length() > 0)
1171 {
1172 populationSize = Integer.parseInt(value);
1173 }
1174 break;
1175 }
1176
1177 case "NUMCONVGEN=":
1178 {
1179 if (value.length() > 0)
1180 {
1181 numConvGen = Integer.parseInt(value);
1182 }
1183 break;
1184 }
1185
1186 case "KEEPNEWRINGSYSTEMVERTEXES":
1187 {
1188 saveRingSystemsAsTemplatesNonScaff = true;
1189 break;
1190 }
1191
1192 case "KEEPNEWRINGSYSTEMSCAFFOLDS":
1193 {
1194 saveRingSystemsAsTemplatesScaffolds = true;
1195 break;
1196 }
1197
1198
1199 case "KEEPNEWRINGSYSTEMFITNESSTRSH=":
1200 {
1201 if (value.length() > 0)
1202 {
1203 saveRingSystemsFitnessThreshold = Double.parseDouble(value);
1204 }
1205 break;
1206 }
1207
1208 case "MULTISITEMUTATIONWEIGHTS=":
1209 {
1210 String[] ws = value.split(",|\\s+");
1211 List<Double> lst = new ArrayList<Double>();
1212 for (String w : ws)
1213 {
1214 if (!w.trim().equals(""))
1215 lst.add(Double.parseDouble(w));
1216 }
1217 mutliSiteMutationWeights = new double[lst.size()];
1218 for (int i=0; i<lst.size(); i++)
1219 {
1220 mutliSiteMutationWeights[i] = lst.get(i);
1221 }
1222 break;
1223 }
1224
1225 case "XOVERSELECTIONMODE=":
1226 {
1227 if (value.length() > 0)
1228 {
1229 xoverSelectionMode = -1;
1230 if (value.compareToIgnoreCase("TS") == 0)
1231 {
1232 xoverSelectionMode = 1;
1233 strXoverSelectionMode = "TOURNAMENT";
1234 } else if (value.compareToIgnoreCase("RW") == 0)
1235 {
1236 xoverSelectionMode = 2;
1237 strXoverSelectionMode = "ROULETTE WHEEL";
1238 } else if (value.compareToIgnoreCase("SUS") == 0)
1239 {
1240 xoverSelectionMode = 3;
1241 strXoverSelectionMode = "STOCHASTIC UNIVERSAL SAMPLING";
1242 } else if (value.compareToIgnoreCase("RANDOM") == 0)
1243 {
1244 xoverSelectionMode = 4;
1245 strXoverSelectionMode = "RANDOM";
1246 } else {
1247 throw new DENOPTIMException("ERROR! String '"
1248 + value + "' cannot be interpreted as a "
1249 + "parent selection strategy.");
1250 }
1251 }
1252 break;
1253 }
1254
1255 case "NUMOFFSPRINGFROMXOVER=":
1256 {
1257 if (value.length() > 0)
1258 {
1259 maxOffsprintFromXover = Integer.parseInt(value);
1260 if (maxOffsprintFromXover>2)
1261 throw new DENOPTIMException("ERROR! Can only generate "
1262 + "up to 2 offspring from crossover, but you "
1263 + "required " + maxOffsprintFromXover);
1264 }
1265 break;
1266 }
1267
1268 case "KEEPBESTSIBLING=":
1269 {
1270 keepBestSibling = readYesNoTrueFalse(value);
1271 break;
1272 }
1273
1274 case "MUTATEDGRAPHCHECKFAILTOLERANT=":
1275 {
1276 mutatedGraphFailedEvalTolerant = readYesNoTrueFalse(value);
1277 break;
1278 }
1279
1280 case "XOVERGRAPHCHECKFAILTOLERANT=":
1281 {
1282 xoverGraphFailedEvalTolerant = readYesNoTrueFalse(value);
1283 break;
1284 }
1285
1286 case "MUTATIONFAILURETOLERANT=":
1287 {
1288 mutationFailureTolerant = readYesNoTrueFalse(value);
1289 break;
1290 }
1291
1292 case "XOVERFAILURETOLERANT=":
1293 {
1294 xoverFailureTolerant = readYesNoTrueFalse(value);
1295 break;
1296 }
1297
1298 case "BUILDFAILURETOLERANT=":
1299 {
1300 buildAnewFailureTolerant = readYesNoTrueFalse(value);
1301 break;
1302 }
1303
1304 case "MAXXOVERSUBGRAPHSIZE=":
1305 {
1306 if (value.length() > 0)
1307 {
1308 maxXOverableSubGraphSize = Integer.parseInt(value);
1309 }
1310 break;
1311 }
1312
1313 case "WRITEPOPULATIONTOFILE":
1314 {
1315 writePopOnDisk = true;
1316 break;
1317 }
1318
1319 default:
1320 msg = "Keyword " + key + " is not a known GeneticAlgorithm-"
1321 + "related keyword. Check input files.";
1322 throw new DENOPTIMException(msg);
1323 }

1328 public static int convertProbabilityScheme(String option)
1329 throws DENOPTIMException
1330 {
