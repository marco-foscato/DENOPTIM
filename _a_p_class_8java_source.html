275 private static APClass getUnique(String rule, int subClass, BondType bt)
276 {
277 APClass newApc = new APClass();
278 synchronized (uniqueAPClassesLock)
279 {
280 boolean found = false;
281 for (APClass existingApc : uniqueAPClasses)
282 {
283 if (existingApc.getRule().equals(rule)
284 && existingApc.getSubClass()==subClass)
285 {
286 newApc = existingApc;
287 found = true;
288 break;
289 }
290 }
291 if (!found)
292 {
293 newApc.setRule(rule);
294 newApc.setSubClass(subClass);
295 newApc.setBondType(bt);
296 } else {
297 // NB: the default bond type for RCAs must not be changed, but
298 // For non-RCA APClasses we do update the bond type.
299 if (bt != newApc.bndTyp && !RCAAPCLASSSET.contains(newApc))
300 {
301 System.err.println("WARNING! Changing bond order of "
302 + "APClass " + newApc + ": " + newApc.bndTyp
303 + " -> " + bt);
304 newApc.setBondType(bt);
305 }
306 }
307 uniqueAPClasses.add(newApc);
308 }
309 return newApc;
310 }

445 public static boolean isValidAPClassString(String s)
446 {
447 if (!s.matches("^[a-z,A-Z,0-9].*"))
448 return false;
449
450 if (!s.matches(".*[0-9]$"))
451 return false;
452
453 if (s.contains(" "))
454 return false;
455
456 if (!s.contains(DENOPTIMConstants.SEPARATORAPPROPSCL))
457 return false;
458
459 int numSep = 0;
460 for (int i=0; i<s.length(); i++)
461 {
462 if (s.charAt(i) == DENOPTIMConstants.SEPARATORAPPROPSCL.charAt(0))
463 {
464 numSep++;
465 }
466 }
467 if (numSep != 1)
468 return false;
469
470 String apRule = s.split(DENOPTIMConstants.SEPARATORAPPROPSCL)[0];
471 String sub = s.split(DENOPTIMConstants.SEPARATORAPPROPSCL)[1];
472
473 return APClass.isValidAPRuleString(apRule)
474 && APClass.isValidAPSubCLassString(sub);
475 }

530 public static class APClassDeserializer
531 implements JsonDeserializer<APClass>
532 {
533 @Override
534 public APClass deserialize(JsonElement json, Type typeOfT,
535 JsonDeserializationContext context) throws JsonParseException
536 {
537 JsonObject jo = json.getAsJsonObject();
538 APClass apc = null;
539 if (jo.has("bndTyp"))
540 {
541 apc = getUnique(jo.get("rule").getAsString(),
542 jo.get("subClass").getAsInt(),
543 context.deserialize(jo.get("bndTyp"),BondType.class));
544 } else {
545 //Only for conversion to V3
546 String rule = jo.get("rule").getAsString();
547 int subClass = jo.get("subClass").getAsInt();
548 boolean found = false;
549 for (APClass existingApc : uniqueAPClasses)
550 {
551 if (existingApc.getRule().equals(rule)
552 && existingApc.getSubClass()==subClass)
553 {
554 apc = existingApc;
555 found = true;
556 break;
557 }
558 }
559 if (!found)
560 {
561 System.out.println("WARNING! Setting " + DEFAULTBT
562 + " for "+rule+":"+subClass);
563 apc = getUnique(jo.get("rule").getAsString(),
564 jo.get("subClass").getAsInt(), DEFAULTBT);
565 }
566 }
567 return apc;
568 }
