130 public List<Ring> getRandomCombinationOfRings(IAtomContainer inMol,
131 DGraph molGraph, int maxRingClosures) throws DENOPTIMException
132 {
133 // Prepare molecular representation with no dummy atoms
134 // NOTE: the connectivity of this molecule is going to be edited
135 // as we identify new candidate DENOPTIMRings. This is done to
136 // calculate ring size as if the DENOPTIMRings were closed
137 IAtomContainer mol;
138 try
139 {
140 mol = (IAtomContainer) inMol.clone();
141 }
142 catch (Throwable t)
143 {
144 throw new DENOPTIMException(t);
145 }
146
147 // Get manager of ring size problems
148 RingSizeManager rsm = new RingSizeManager(fragSpace);
149
150 rsm.initialize(mol, molGraph);
151
152 // Get weighted list of RCVs
153 ArrayList<Vertex> wLstVrtI = rsm.getRSBiasedListOfCandidates();
154
155 // Randomly choose the compatible combinations of RCAs and store them
156 // as DENOPTIMRings.
157 List<Ring> combOfRings = new ArrayList<Ring>();
158 while (wLstVrtI.size() > 0)
159 {
160 // Termination criterion based on maximum number of rings per graph
161 if (combOfRings.size() >= maxRingClosures)
162 break;
163
164 int vIdI = settings.getRandomizer().nextInt(wLstVrtI.size());
165 Vertex vI = wLstVrtI.get(vIdI);
166 wLstVrtI.removeAll(Collections.singleton(vI));
167
168 // Create vector of possible choices for second RCV
169 ArrayList<Vertex> wLstVrtJ = rsm.getRSBiasedListOfCandidates(vI);
170 while (wLstVrtJ.size() > 0)
171 {
172 int vIdJ = settings.getRandomizer().nextInt(wLstVrtJ.size());
173 Vertex vJ = wLstVrtJ.get(vIdJ);
174 wLstVrtJ.removeAll(Collections.singleton(vJ));
175
176 PathSubGraph path = new PathSubGraph(vI,vJ,molGraph);
177 if (evaluatePathClosability(path, inMol))
178 {
179 ArrayList<Vertex> arrLst = new ArrayList<Vertex>();
180 arrLst.addAll(path.getVertecesPath());
181 Ring ring = new Ring(arrLst);
182
183 BondType bndTypI = vI.getEdgeToParent().getBondType();
184 BondType bndTypJ = vJ.getEdgeToParent().getBondType();
185 if (bndTypI != bndTypJ)
186 {
187 String s = "Attempt to close rings is not compatible "
188 + "to the different bond type specified by the "
189 + "head and tail APs: (" + bndTypI + "!="
190 + bndTypJ + " for vertices " + vI + " "
191 + vJ + ")";
192 throw new DENOPTIMException(s);
193 }
194 ring.setBondType(bndTypI);
195
196 combOfRings.add(ring);
197
198 wLstVrtI.removeAll(Collections.singleton(vJ));
199
200 // Update ring sizes according to the newly added bond
201 if (vI instanceof Fragment && vJ instanceof Fragment)
202 {
203 rsm.addRingClosingBond(vI,vJ);
204 }
205 rsm.setVertexAsDone(vJ);
206 break;
207 }
208 }
209 rsm.setVertexAsDone(vI);
210 }
211
212 logger.log(Level.FINE,"Random combOfRings: "+combOfRings);
213
214 return combOfRings;
215 }

227 public ArrayList<List<Ring>> getPossibleCombinationOfRings(
228 IAtomContainer mol, DGraph molGraph)
229 throws DENOPTIMException
230 {
231 // All the candidate paths
232 Map<ObjectPair,PathSubGraph> allGoodPaths =
233 new HashMap<ObjectPair,PathSubGraph>();
234 ArrayList<Vertex> rcaVertLst = molGraph.getFreeRCVertices();
235
236 // Get manager of ring size problems
237 RingSizeManager rsm = new RingSizeManager(fragSpace);
238 rsm.initialize(mol, molGraph);
239
240 // identify compatible pairs of RCA vertices
241 Map<Vertex,ArrayList<Vertex>> compatMap =
242 new HashMap<Vertex,ArrayList<Vertex>>();
243 for (int i=0; i<rcaVertLst.size(); i++)
244 {
245 Vertex vI = rcaVertLst.get(i);
246 for (int j=i+1; j<rcaVertLst.size(); j++)
247 {
248 Vertex vJ = rcaVertLst.get(j);
249 if (!rsm.getCompatibilityOfPair(vI,vJ))
250 {
251 logger.log(Level.FINE, "Rejecting RC-incompatible pair "
252 + vI + " "+ vJ);
253 continue;
254 }
255
256 // make the new candidate RCA pair
257 PathSubGraph subGraph = new PathSubGraph(vI, vJ, molGraph);
258 logger.log(Level.FINE, "Evaluating closability of path "
259 + subGraph);
260 boolean keepRcaPair = evaluatePathClosability(subGraph, mol);
261
262 if (!keepRcaPair)
263 {
264 logger.log(Level.FINE, "Rejecting RCA pair");
265 continue;
266 }
267
268 // finally store this pair as a compatible pair
269 logger.log(Level.FINE, "All compatibility criteria satisfied: "
270 + "Storing verified RCA pair");
271
272 // Store the information that the two vertex are compatible
273 if (compatMap.containsKey(vI))
274 {
275 compatMap.get(vI).add(vJ);
276 }
277 else
278 {
279 ArrayList<Vertex> lst =
280 new ArrayList<Vertex>();
281 lst.add(vJ);
282 compatMap.put(vI,lst);
283 }
284 if (compatMap.containsKey(vJ))
285 {
286 compatMap.get(vJ).add(vI);
287 }
288 else
289 {
290 ArrayList<Vertex> lst =
291 new ArrayList<Vertex>();
292 lst.add(vI);
293 compatMap.put(vJ,lst);
294 }
295
296 // store the RCA pair for further use
297 ObjectPair compatPair;
298 if (vI.getVertexId() > vJ.getVertexId())
299 {
300 compatPair = new ObjectPair(vI,vJ);
301 }
302 else
303 {
304 compatPair = new ObjectPair(vJ,vI);
305 }
306 allGoodPaths.put(compatPair,subGraph);
307 }
308 }
309
310 logger.log(Level.FINE, "Compatibility Map for RCAs: "+NL+compatMap);
311
312 // Identify paths that share bonds (interdependent paths)
313 Map<IBond,List<PathSubGraph>> interdepPaths =
314 new HashMap<IBond,List<PathSubGraph>>();
315 if (settings.checkInterdependentChains())
316 {
317 for (PathSubGraph rpA : allGoodPaths.values())
318 {
319 for (PathSubGraph rpB : allGoodPaths.values())
320 {
321 if (rpA == rpB)
322 {
323 continue;
324 }
325
326 Vertex hA = rpA.getHeadVertex();
327 Vertex tA = rpA.getTailVertex();
328 Vertex hB = rpB.getHeadVertex();
329 Vertex tB = rpB.getTailVertex();
330
331 if ((hA == hB || hA == tB) || (tA == hB || tA == tB))
332 {
333 continue;
334 }
335
336 for (IBond bnd : rpA.getBondPath())
337 {
338 // ignore non-rotatable bonds
339 // NOTE: here we assume that rotatable bonds have been
340 // identified before.
341 Object rotFlg = bnd.getProperty(
342 DENOPTIMConstants.BONDPROPROTATABLE);
343 if (rotFlg==null || !Boolean.valueOf(rotFlg.toString()))
344 {
345 continue;
346 }
347
348 if (rpB.getBondPath().contains(bnd))
349 {
350 if (interdepPaths.containsKey(bnd))
351 {
352 interdepPaths.get(bnd).add(rpA);
353 interdepPaths.get(bnd).add(rpB);
354 }
355 else
356 {
357 List<PathSubGraph> paths =
358 new ArrayList<PathSubGraph>();
359 paths.add(rpA);
360 paths.add(rpB);
361 interdepPaths.put(bnd,paths);
362 }
363 }
364 }
365 }
366 }
367 StringBuilder sb = new StringBuilder();
368 for (IBond bnd : interdepPaths.keySet())
369 {
370 sb.append("Interdependent paths for bond " + bnd);
371 List<PathSubGraph> sop = interdepPaths.get(bnd);
372 for (PathSubGraph p : sop)
373 {
374 sb.append(NL + " -> vA: "+p.getHeadVertex()
375 +" vB: "+p.getTailVertex());
376 }
377 }
378 logger.log(Level.FINE, sb.toString());
379 }
380
381 // Generate all combinations of compatible, closable paths
382 ArrayList<ObjectPair> lstPairs = new ArrayList<ObjectPair>();
383 ArrayList<Long> usedId = new ArrayList<Long>();
384 ArrayList<Vertex> sortedKeys = new ArrayList<Vertex>();
385 for (Vertex keyVert : compatMap.keySet())
386 {
387 sortedKeys.add(keyVert);
388 }
389
390 // All possible ring closing paths will be stored here
391 ArrayList<List<Ring>> allCombsOfRings =
392 new ArrayList<List<Ring>>();
393 combineCompatPathSubGraphs(0,
394 sortedKeys,
395 compatMap,
396 lstPairs,
397 usedId,
398 allGoodPaths,
399 interdepPaths,
400 allCombsOfRings);
401 logger.log(Level.FINE, "All possible combination of rings: " +
402 allCombsOfRings);
403 return allCombsOfRings;
404 }

412 private boolean combineCompatPathSubGraphs(int ii0,
413 ArrayList<Vertex> sortedKeys,
414 Map<Vertex,ArrayList<Vertex>> compatMap,
415 ArrayList<ObjectPair> lstPairs,
416 ArrayList<Long> usedId,
417 Map<ObjectPair,PathSubGraph> allGoodPaths,
418 Map<IBond,List<PathSubGraph>> interdepPaths,
419 ArrayList<List<Ring>> allCombsOfRings)
420 throws DENOPTIMException
421 {
422 int objId = this.hashCode();
423 String recLab = new String(new char[recCount]).replace("\0", "-");
424
425 StringBuilder sb = new StringBuilder();
426 sb.append(objId+"-"+recLab+"> Begin of new recursion: "+recCount+NL);
427 sb.append(objId+"-"+recLab+"> sortedKeys= "+sortedKeys+NL);
428 sb.append(objId+"-"+recLab+"> usedId= "+usedId+NL);
429 sb.append(objId+"-"+recLab+"> ii0= "+ii0+NL);
430 sb.append(objId+"-"+recLab+"> lstPairs= "+lstPairs+NL);
431 sb.append(objId+"-"+recLab+"> compatMap= "+compatMap+NL);
432 sb.append(objId+"-"+recLab+"> allCombsOfRings"+NL);
433 for (List<Ring> ringSet : allCombsOfRings)
434 {
435 sb.append(" "+ringSet+NL);
436 }
437 logger.log(Level.FINEST, sb.toString());
438
439 boolean inFound = false;
440 boolean addedNew = false;
441 for (int ii=ii0; ii<sortedKeys.size(); ii++)
442 {
443 Vertex vi = sortedKeys.get(ii);
444 long vIdI = vi.getVertexId();
445
446 logger.log(Level.FINEST, objId+"-"+recLab+"> vIdI= "+vIdI);
447
448 if (usedId.contains(vIdI))
449 {
450 continue;
451 }
452
453 for (Vertex vj : compatMap.get(vi))
454 {
455 long vIdJ = vj.getVertexId();
456
457 logger.log(Level.FINEST, objId+"-"+recLab+"> vIdJ= "+vIdJ);
458
459 if (usedId.contains(vIdJ) || usedId.contains(vIdI))
460 {
461 continue;
462 }
463
464 ObjectPair op;
465 if (vIdI > vIdJ)
466 {
467 op = new ObjectPair(vi,vj);
468 }
469 else
470 {
471 op = new ObjectPair(vj,vi);
472 }
473 lstPairs.add(op);
474 usedId.add(vIdI);
475 usedId.add(vIdJ);
476
477 if (lstPairs.size() > maxLng)
478 {
479 maxLng = lstPairs.size();
480 }
481 recCount++;
482 inFound = combineCompatPathSubGraphs(ii+1,
483 sortedKeys,
484 compatMap,
485 lstPairs,
486 usedId,
487 allGoodPaths,
488 interdepPaths,
489 allCombsOfRings);
490 recCount--;
491
492 logger.log(Level.FINEST, objId+"-"+recLab+"> lstPairs.size() "
493 + "& maxLng= "+lstPairs.size() +" " +maxLng);
494
495 if (!inFound && lstPairs.size() == maxLng)
496 {
497 logger.log(Level.FINEST, objId+"-"+recLab+"> in A");
498
499 boolean closable = true;
500 if (settings.checkInterdependentChains() &&
501 hasInterdependentPaths(lstPairs, interdepPaths))
502 {
503 closable = checkClosabilityOfInterdependentPaths(
504 lstPairs,
505 interdepPaths,
506 allGoodPaths);
507 }
508 if (closable)
509 {
510 logger.log(Level.FINEST, objId+"-"+recLab+"> in B");
511
512 List<Ring> ringsComb = new ArrayList<Ring>();
513 for (ObjectPair opFinal : lstPairs)
514 {
515 PathSubGraph path = allGoodPaths.get(opFinal);
516 ArrayList<Vertex> arrLst =
517 new ArrayList<Vertex>();
518 arrLst.addAll(path.getVertecesPath());
519
520 Ring ring = new Ring(arrLst);
521
522 List<Edge> es = path.getEdgesPath();
523 BondType btH = es.get(0).getBondType();
524 BondType btT = es.get(es.size()-1).getBondType();
525 if (btH != btT)
526 {
527 String s = "Attempt to close rings is not "
528 + "compatible to the different bond type "
529 + "specified by the head and tail APs: ("
530 + btH + "!=" + btT + " for vertices "
531 + path.getHeadVertex() + " "
532 + path.getTailVertex() + ")";
533 throw new DENOPTIMException(s);
534 }
535 ring.setBondType(btH);
536
537 ringsComb.add(ring);
538
539 logger.log(Level.FINEST, objId+"-"+recLab
540 +"> added ringComb: "+ring);
541 }
542
543 boolean notNewCmb = false;
544 for(List<Ring> oldCmb : allCombsOfRings)
545 {
546 StringBuilder sb2 = new StringBuilder();
547 sb2.append(objId+"-"+recLab
548 +"> Comparing ring sets: ");
549 sb2.append("o> old: "+oldCmb);
550 sb2.append("o> new: "+ringsComb);
551 logger.log(Level.FINEST, sb2.toString());
552
553 notNewCmb = RingClosingUtils.areSameRingsSet(oldCmb,
554 ringsComb);
555
556 logger.log(Level.FINEST,"o> result: "+notNewCmb);
557
558 if (notNewCmb)
559 {
560 break;
561 }
562 }
563
564 if (!notNewCmb)
565 {
566 logger.log(Level.FINEST, objId+"-"+recLab
567 +"> adding to all combs of ring.");
568 allCombsOfRings.add(ringsComb);
569 addedNew = true;
570 }
571 }
572 }
573 if (!inFound)
574 {
575 logger.log(Level.FINEST, objId+"-"+recLab+"> in C");
576
577 ArrayList<ObjectPair> toDel = new ArrayList<ObjectPair>();
578 for (int ir = recCount; ir<lstPairs.size(); ir++)
579 {
580 ObjectPair opToRemove = lstPairs.get(ir);
581 Vertex delVA = (Vertex) opToRemove.getFirst();
582 Vertex delVB = (Vertex) opToRemove.getSecond();
583 usedId.remove(usedId.indexOf(delVA.getVertexId()));
584 usedId.remove(usedId.indexOf(delVB.getVertexId()));
585 toDel.add(opToRemove);
586 }
587 lstPairs.removeAll(toDel);
588
589 logger.log(Level.FINEST, objId+"-"+recLab
590 +"> in C: after removal usedId: "+usedId+NL
591 +objId+"-"+recLab
592 +"> in C: after removal lstPairs: "+lstPairs);
593 }
594
595 if (lstPairs.contains(op))
596 {
597 logger.log(Level.FINEST, objId+"-"+recLab+"> in D");
598
599 lstPairs.remove(op);
600 usedId.remove(usedId.indexOf(vIdI));
601 usedId.remove(usedId.indexOf(vIdJ));
602
603 logger.log(Level.FINEST, objId+"-"+recLab
604 +"> in D: after removal usedId: "+usedId
605 +objId+"-"+recLab
606 +"> in D: after removal lstPairs: "+lstPairs);
607 }
608 }
609 }
610
611 logger.log(Level.FINEST, objId+"-"+recLab+"> returning= "+addedNew);
612
613 return addedNew;
614 }

623 private boolean hasInterdependentPaths(ArrayList<ObjectPair> lstPairs,
624 Map<IBond,List<PathSubGraph>> interdepPaths)
625 {
626 boolean result = false;
627
628 for (IBond bnd : interdepPaths.keySet())
629 {
630 List<PathSubGraph> psgSet = interdepPaths.get(bnd);
631 for (PathSubGraph psg : psgSet)
632 {
633 Vertex va1 = psg.getHeadVertex();
634 Vertex va2 = psg.getTailVertex();
635
636 for (ObjectPair op : lstPairs)
637 {
638 Vertex vb1 = (Vertex) op.getFirst();
639 Vertex vb2 = (Vertex) op.getSecond();
640
641 if ((va1 == vb1 && va2 == vb2) ||
642 (va1 == vb2 && va2 == vb1))
643 {
644 result = true;
645 break;
646 }
647 }
648 if (result)
649 break;
650 }
651 if (result)
652 break;
653 }
654 return result;
655 }

665 private boolean checkClosabilityOfInterdependentPaths(
666 ArrayList<ObjectPair> lstPairs,
667 Map<IBond,List<PathSubGraph>> interdepPaths,
668 Map<ObjectPair,PathSubGraph> allGoodPaths)
669 {
670 // Identify the interdependent sets of paths
671 List<ArrayList<ObjectPair>> listOfIntrDepPaths =
672 new ArrayList<ArrayList<ObjectPair>>();
673 for (IBond bnd : interdepPaths.keySet())
674 {
675 ArrayList<ObjectPair> locSop = new ArrayList<ObjectPair>();
676 List<PathSubGraph> psgSet = interdepPaths.get(bnd);
677 for (PathSubGraph psg : psgSet)
678 {
679 Vertex va1 = psg.getHeadVertex();
680 Vertex va2 = psg.getTailVertex();
681
682 for (ObjectPair op : lstPairs)
683 {
684 Vertex vb1 = (Vertex) op.getFirst();
685 Vertex vb2 = (Vertex) op.getSecond();
686
687 if ((va1 == vb1 && va2 == vb2) ||
688 (va1 == vb2 && va2 == vb1))
689 {
690 locSop.add(op);
691 }
692 }
693 }
694 if (locSop.size() > 1)
695 {
696 listOfIntrDepPaths.add(locSop);
697 }
698 }
699
700 // Verify simultaneous closeness of each group of interdependent paths
701 boolean closable = true;
702 for (ArrayList<ObjectPair> grpIntrdepPaths : listOfIntrDepPaths)
703 {
704 // Per each closable conf. of the first path stores the paths
705 // that have a simultaneously closable conf.
706 Map<ClosableConf,List<ObjectPair>> mapOcPathsWithCC =
707 new HashMap<ClosableConf,List<ObjectPair>>();
708
709 // Per each closable conf. of a path lists the simultaneously
710 // closable confs of the other interdependent paths
711 Map<ClosableConf,List<ClosableConf>> mapOfClosableConfs =
712 new HashMap<ClosableConf,List<ClosableConf>>();
713
714 // For the first path just put all closable conf in the list
715 ObjectPair firstOp = grpIntrdepPaths.get(0);
716 PathSubGraph firstPsg = allGoodPaths.get(firstOp);
717 RingClosingConformations firstRcc = firstPsg.getRCC();
718
719 for (ArrayList<Double> ccAngls : firstRcc.getListOfConformations())
720 {
721 ClosableConf cc = new ClosableConf(firstPsg.getBondPath(),
722 ccAngls);
723 // Container for cc of other paths
724 List<ClosableConf> scc = new ArrayList<ClosableConf>();
725 mapOfClosableConfs.put(cc,scc);
726 // Container for other paths
727 List<ObjectPair> sop = new ArrayList<ObjectPair>();
728 sop.add(firstOp);
729 mapOcPathsWithCC.put(cc,sop);
730 }
731
732 // For other paths put only the closable confs (CC) that are
733 // simultaneously closable with previously listed CCs
734 for (int iOp=1; iOp<grpIntrdepPaths.size(); iOp++)
735 {
736 ObjectPair locOp = grpIntrdepPaths.get(iOp);
737 PathSubGraph locPath = allGoodPaths.get(locOp);
738 RingClosingConformations rccOfLocPath = locPath.getRCC();
739
740 // try with each closable conf of the local path
741 for (ArrayList<Double> ccAngles :
742 rccOfLocPath.getListOfConformations())
743 {
744 ClosableConf locCC = new ClosableConf(
745 locPath.getBondPath(), ccAngles);
746 // Compare the locCC with CC of the first path
747 for (ClosableConf frstCC : mapOfClosableConfs.keySet())
748 {
749 if (frstCC.shareBond(locCC) &&
750 frstCC.canCoexistWith(locCC))
751 {
752 // CC of loc path is simultaneously closable with
753 // this CC of first path (frstCC)
754 if (mapOfClosableConfs.get(frstCC).size() == 0)
755 {
756 // the list of CC simultaneously closable with
757 // the key CC (frstCC) is empty, so just
758 // add locCC for future needs
759 mapOfClosableConfs.get(frstCC).add(locCC);
760 // add to Set wont add if is already there
761 mapOcPathsWithCC.get(frstCC).add(locOp);
762 }
763 else
764 {
765 // the list of CC simultaneously closable with
766 // the key CC (frstCC) has already one or more
767 // entries: we need to compare this locCC with
768 // all these entries
769 boolean canCoexistWithAll = true;
770 for (ClosableConf lstCC :
771 mapOfClosableConfs.get(frstCC))
772 {
773 if (lstCC.shareBond(locCC) &&
774 !lstCC.canCoexistWith(locCC))
775 {
776 // Sorry, locCC cannot coexist
777 canCoexistWithAll = false;
778 break;
779 }
780 }
781 if (canCoexistWithAll)
782 {
783 mapOfClosableConfs.get(frstCC).add(locCC);
784 // add to Set wont add if is already there
785 mapOcPathsWithCC.get(frstCC).add(locOp);
786 }
787 }
788 }
789 }
790 }
791
792 boolean goon = true;
793 for (ClosableConf frstCC : mapOfClosableConfs.keySet())
794 {
795 if (!mapOcPathsWithCC.get(frstCC).contains(locOp))
796 {
797 goon = false;
798 break;
799 }
800 }
801 if (!goon)
802 {
803 break;
804 }
805 }
806
807 // Now check if there is at least one set of simultaneously
808 // closable conformations for all members of this group
809 // of interdependent paths
810 boolean foundOneSetFullyClosable = false;
811 for (ClosableConf cc : mapOfClosableConfs.keySet())
812 {
813 int numPathWithSimCC = mapOcPathsWithCC.get(cc).size();

961 private void calculateCompatibilityOfAllRCAPairs() throws DENOPTIMException
962 {
963 weigths = new ArrayList<Double>(Collections.nCopies(sz, 0.0));
964 compatibilityOfPairs = new boolean[sz][sz];
965 for (int i=0; i<sz; i++)
966 {
967 Vertex vI = lstVert.get(i);
968 IAtom atmI = null;
969 RingClosingAttractor rcaI = null;
970 boolean isAtmI = vI instanceof Fragment;
971 if (isAtmI)
972 {
973 atmI = mol.getAtom(vIdToAtmId.get(vI).get(0));
974
975 // Dealing with the possibility that RCV is the scaffold
976 if (mol.getConnectedAtomsList(atmI).size()==0)
977 continue;
978
979 rcaI = new RingClosingAttractor(atmI,mol);
980 if (!rcaI.isAttractor())
981 {
982 String s = "Attempt to evaluate RCA pair compatibility "
983 + "with a non-RCA end (" + atmI + ").";
984 throw new DENOPTIMException(s);
985 }
986 }
987
988 for (int j=i+1; j<sz; j++)
989 {
990 Vertex vJ = lstVert.get(j);
991 IAtom atmJ = null;
992 RingClosingAttractor rcaJ = null;
993 boolean isAtmJ = vJ instanceof Fragment;
994 if (isAtmJ)
995 {
996 atmJ = mol.getAtom(vIdToAtmId.get(vJ).get(0));
997
998 rcaJ = new RingClosingAttractor(atmJ,mol);
999 if (!rcaJ.isAttractor())
1000 {
1001 String s = "Attempt to evaluate RCA pair compatibility "
1002 + "with a non-RCA end (" + atmJ + ").";
1003 throw new DENOPTIMException(s);
1004 }
1005 }
1006
1007 if ((isAtmI && !isAtmJ) || (!isAtmI && isAtmJ))
1008 {
1009 continue;
1010 }
1011 if (!isAtmI && !isAtmJ)
1012 {
1013 // Ring size is ignored when RCVs are empty vertexes
1014 compatibilityOfPairs[i][j] = true;
1015 compatibilityOfPairs[j][i] = true;
1016 weigths.set(i, weigths.get(i) + 1.0);
1017 weigths.set(j, weigths.get(j) + 1.0);
1018 continue;
1019 }
1020
1021 if (rcaI.isCompatible(rcaJ)
1022 && evaluateRCVPair(vI,vJ,fragSpace))
1023 {
1024 //TODO: evaluate the use of ShortestPath instead of this
1025 int ringSize = topoMat[vIdToAtmId.get(vI).get(0)]
1026 [vIdToAtmId.get(vJ).get(0)] - 1;
1027 int szFct = 0;
1028 if (ringSize < settings.getMaxRingSize())
1029 {
1030 szFct = ringSizeBias.get(ringSize);
1031 }
1032 if (szFct > 0)
1033 {
1034 compatibilityOfPairs[i][j] = true;
1035 compatibilityOfPairs[j][i] = true;
1036 weigths.set(i, weigths.get(i) + szFct);
1037 weigths.set(j, weigths.get(j) + szFct);
1038 }
1039
1040 logger.log(Level.FINE, " i:" + i + " j:" + j + " size:"
1041 + ringSize + " factors:" + weigths);
1042 }
1043 }
1044 }
1045
1046 if (logger.isLoggable(Level.FINE))
1047 {
1048 StringBuilder sb = new StringBuilder();
1049 sb.append("RCV pairs compatibility (ring size-biased):"+NL);
1050 for (int i=0; i<sz;i++)
1051 {
1052 String l = " ";
1053 for (int j=0; j<sz; j++)
1054 {
1055 String p = "0";
1056 if (compatibilityOfPairs[i][j])
1057 p = "1";
1058 l = l + " " + p;
1059 }
1060 sb.append(l+NL);
1061 }
1062 logger.log(Level.FINE, sb.toString());
1063 }

1299 private boolean evaluateRCVPair(Vertex vI, Vertex vJ,
1300 FragmentSpace fragSpace)
1301 throws DENOPTIMException
1302 {
1303 String s = "Evaluation of RCV pair " + vI + " " + vJ + ": ";
1304
1305 // Get details on the first vertex (head)
1306 Edge edgeI = vI.getEdgeToParent();
1307 int srcApIdI = edgeI.getSrcAPID();
1308 Vertex pvI = vI.getParent();
1309 AttachmentPoint srcApI = pvI.getAttachmentPoints().get(srcApIdI);
1310 int srcAtmIdI = srcApI.getAtomPositionNumber();
1311 APClass parentAPClsI = edgeI.getSrcAPClass();
1312
1313 // Get details on the second vertex (tail)
1314 Edge edgeJ = vJ.getEdgeToParent();
1315 int srcApIdJ = edgeJ.getSrcAPID();
1316 Vertex pvJ = vJ.getParent();
1317 AttachmentPoint srcApJ =pvJ.getAttachmentPoints().get(srcApIdJ);
1318 int srcAtmIdJ = srcApJ.getAtomPositionNumber();
1319 APClass parentAPClsJ = edgeJ.getSrcAPClass();
1320
1321 // exclude if no entry in RC-Compatibility map
1322 if (!fragSpace.getRCCompatibilityMatrix().containsKey(parentAPClsI))
1323 {
1324 logger.log(Level.FINE, s + "RC-CPMap does not contain class (I) "
1325 + parentAPClsI + " " + parentAPClsI.hashCode());
1326 return false;
1327 }
1328 ArrayList<APClass> compatClassesI = fragSpace.getRCCompatibilityMatrix()
1329 .get(parentAPClsI);
1330
1331 // exclude if no entry in RC-Compatibility map
1332 if (!fragSpace.getRCCompatibilityMatrix().containsKey(parentAPClsJ))
1333 {
1334 logger.log(Level.FINE, s + "RC-CPMap does not contain class (J) "
1335 + parentAPClsJ + " " + parentAPClsJ.hashCode());
1336 return false;
1337 }
1338 ArrayList<APClass> compatClassesJ = fragSpace.getRCCompatibilityMatrix()
1339 .get(parentAPClsJ);
1340
1341 // exclude loops included within a single vertex
1342 if (vI == vJ)
1343 {
1344 logger.log(Level.FINE, s + "vI same as vJ: loop not allowed!");
1345 return false;
1346 }
1347
1348 // exclude pairs of RCA-vertices having same src atom
1349 if (pvI == pvJ && srcAtmIdI == srcAtmIdJ)
1350 {
1351 logger.log(Level.FINE, s + "Same src: " + pvI + " " + pvJ
1352 + " " + srcAtmIdI + " " + srcAtmIdJ);
1353 return false;
1354 }
1355
1356 // exclude paths that do not connect APClass compatible ends
1357 // NOTE that in ring closures the CPMap is symmetric, this
1358 // also implies that CPMap for ring closure may be different
1359 // from standard CPMap
1360 if (!(compatClassesI.contains(parentAPClsJ) ||
1361 compatClassesJ.contains(parentAPClsI)))
1362 {
1363 logger.log(Level.FINE,s + "APClass not compatible "
1364 + parentAPClsJ);
1365 return false;
1366 }
1367
1368 logger.log(Level.FINE, s + "all criteria satisfied.");
1369
1370 return true;
1371 }

1422 private boolean evaluateConstitutionalClosability(PathSubGraph subGraph,
1423 IAtomContainer inMol) throws DENOPTIMException
1424 {
1425 settings.getLogger().log(Level.FINE, "Evaluating constitutional "
1426 + "closability of path: " + subGraph.getVertecesPath());
1427
1428 // Get a working copy of the molecular container
1429 IChemObjectBuilder builder = SilentChemObjectBuilder.getInstance();
1430 IAtomContainer mol = builder.newAtomContainer();
1431 try
1432 {
1433 mol = inMol.clone();
1434 }
1435 catch (CloneNotSupportedException e)
1436 {
1437 throw new DENOPTIMException(e);
1438 }
1439 MoleculeUtils.removeRCA(mol);
1440
1441 // Identify atoms of molecular representation that correspond to
1442 // this path of vertices
1443 Map<Vertex,ArrayList<Integer>> vIdToAtmId =
1444 MoleculeUtils.getVertexToAtomIdMap(
1445 (ArrayList<Vertex>) subGraph.getVertecesPath(),
1446 mol);
1447 List<Integer> atmIdsInVrtxPath = new ArrayList<Integer>();
1448 for (Vertex v : subGraph.getVertecesPath())
1449 {
1450 atmIdsInVrtxPath.addAll(vIdToAtmId.get(v));
1451 }
1452
1453 // Find atoms to remove: keep only the atoms that belong to the
1454 // vertex path and their neighbours
1455 ArrayList<IAtom> toRemove = new ArrayList<IAtom>();
1456 for (int i=0; i<mol.getAtomCount(); i++)
1457 {
1458 if (!atmIdsInVrtxPath.contains(i))
1459 {
1460 IAtom candAtm = mol.getAtom(i);
1461 boolean isNeighbour = false;
1462 List<IAtom> nbrs = mol.getConnectedAtomsList(candAtm);
1463 for (IAtom nbrAtm : nbrs)
1464 {
1465 if (atmIdsInVrtxPath.contains(mol.indexOf(nbrAtm)))
1466 {
1467 isNeighbour = true;
1468 break;
1469 }
1470 }
1471 if (!isNeighbour)
1472 {
1473 toRemove.add(candAtm);
1474 }
1475 }
1476 }
1477 // Deal with RCAs: head and tail vertices
1478 IAtom atmH = mol.getAtom(vIdToAtmId.get(subGraph.getHeadVertex()).get(0));
1479 IAtom atmT = mol.getAtom(vIdToAtmId.get(subGraph.getTailVertex()).get(0));
1480 IAtom srcH = mol.getConnectedAtomsList(atmH).get(0);
1481 IAtom srcT = mol.getConnectedAtomsList(atmT).get(0);
1482 int iSrcH = mol.indexOf(srcH);
1483 int iSrcT = mol.indexOf(srcT);
1484 toRemove.add(atmH);
1485 toRemove.add(atmT);
1486
1487 BondType bndTyp = subGraph.getEdgesPath().get(0).getBondType();
1488 if (bndTyp.hasCDKAnalogue())
1489 {
1490 mol.addBond(iSrcH, iSrcT, bndTyp.getCDKOrder());
1491 } else {
1492 settings.getLogger().log(Level.WARNING,
1493 "WARNING! Attempt to add ring closing bond "
1494 + "did not add any actual chemical bond because the "
1495 + "bond type of the chord is '" + bndTyp +"'.");
1496 }
1497
1498 // Remove atoms
1499 for (IAtom a : toRemove)
1500 {
1501 mol.removeAtom(a);
1502 }
1503
1504 StringBuilder sb = new StringBuilder();
1505 sb.append("Molecular representation of path includes:");
1506 for (IAtom a : mol.atoms())
1507 {
1508 sb.append(" " + a.getSymbol() + mol.indexOf(a) + " "
1509 + a.getProperties());
1510 }
1511 settings.getLogger().log(Level.FINEST, sb.toString());
1512
1513 boolean closable = false;
1514
1515 // Evaluate requirement based on elements contained in the ring
1516 boolean spanRequiredEls = false;
1517 Set<String> reqRingEl = settings.getRequiredRingElements();
1518 if (reqRingEl.size() != 0)
1519 {
1520 // Prepare shortest atom path
1521 List<IAtom> atomsPath = new ArrayList<IAtom>();
1522 try {
1523 ShortestPaths sp = new ShortestPaths(mol, srcH);
1524 atomsPath = new ArrayList<IAtom>(Arrays.asList(sp.atomsTo(srcT)));
1525 } catch (Throwable t) {
1526 throw new DENOPTIMException("PathTools Exception: " + t);
1527 }
1528
1529 // Look for the required elements
1530 String missingEl = "";
1531 for (String el : reqRingEl)
1532 {
1533 missingEl = el;
1534 for (IAtom a : atomsPath)
1535 {
1536 if (a.getSymbol().equals(el))
1537 {
1538 spanRequiredEls = true;
1539 break;
1540 }
1541 }
1542 if (!spanRequiredEls)
1543 {
1544 break;
1545 }
1546 }
1547 if (!spanRequiredEls)
1548 {
1549 settings.getLogger().log(Level.FINER,
1550 "Candidate ring doesn't involve " + missingEl);
1551 return false;
1552 }
1553 closable = true;
1554 }
1555
1556 // Try to find a match for any of the SMARTS queries
1557 Map<String,String> smarts = settings.getConstitutionalClosabilityConds();
1558 if (smarts.size() != 0)
1559 {
1560 closable = false;
1561 ManySMARTSQuery msq = new ManySMARTSQuery(mol,smarts);
1562 if (msq.hasProblems())
1563 {
1564 String msg = "Attempt to match SMARTS for "
1565 + "constitution-based ring-closability conditions "
1566 + "returned an error! Ignoring " + msq.getMessage();
1567 settings.getLogger().log(Level.WARNING,msg);
1568 }
1569 for (String name : smarts.keySet())
1570 {
1571 if (msq.getNumMatchesOfQuery(name) > 0)
1572 {
1573 settings.getLogger().log(Level.FINER,
1574 "Candidate closable path matches constitutional "
1575 + "closability criterion: " + smarts.get(name));
1576 closable = true;
1577 break;
1578 }
1579 }
1580 }
1581
1582 settings.getLogger().log(Level.FINE,
1583 "Contitutional closability: " + closable);
1584 return closable;
1585 }

1681 rca.storeEntry(chainId,closable,rcc);
1682 }

1733 {

1736 isOrphan = true;

1742 logger.log(Level.FINE, "Found orphan: " + vert

1771 }

1783 }

1794 }
1795 }
1796 }
1797 }
