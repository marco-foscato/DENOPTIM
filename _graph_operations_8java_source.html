88 public static List<XoverSite> locateCompatibleXOverPoints(
89 DGraph graphA, DGraph graphB, FragmentSpace fragSpace,
90 int maxSizeXoverSubGraph)
91 throws DENOPTIMException
92 {
93 // First, we identify all the edges that allow crossover, and collect
94 // their target vertexes (i.e., all the potential seed vertexes of
95 // subgraphs that crossover could swap.
96 List<Vertex[]> compatibleVrtxPairs = new ArrayList<Vertex[]>();
97 for (Edge eA : graphA.getEdgeList())
98 {
99 Vertex vA = eA.getTrgAP().getOwner();
100 // We don't do genetic operations on capping vertexes
101 if (vA.getBuildingBlockType() == BBType.CAP)
102 continue;
103
104 for (Edge eB : graphB.getEdgeList())
105 {
106 Vertex vB = eB.getTrgAP().getOwner();
107 // We don't do genetic operations on capping vertexes
108 if (vB.getBuildingBlockType() == BBType.CAP)
109 continue;
110
111 //Check condition for considering this combination
112 if (isCrossoverPossible(eA, eB, fragSpace))
113 {
114 Vertex[] pair = new Vertex[]{vA,vB};
115 compatibleVrtxPairs.add(pair);
116 }
117 }
118 }
119
120 // The crossover sites are the combination of the above compatible
121 // vertexes that define subgraphs respecting the requirements for
122 // being swapped between the two graphs.
123 ArrayList<XoverSite> sites = new ArrayList<XoverSite>();
124 for (Vertex[] pair : compatibleVrtxPairs)
125 {
126 Vertex vA = pair[0];
127 Vertex vB = pair[1];
128 DGraph gA = vA.getGraphOwner();
129 DGraph gB = vB.getGraphOwner();
130
131 // Here we also identify the branch identity of each descendant
132 List<Vertex> descendantsA = new ArrayList<Vertex>();
133 gA.getChildrenTree(vA, descendantsA, true);
134 List<Vertex> descendantsB = new ArrayList<Vertex>();
135 gB.getChildrenTree(vB, descendantsB, true);
136
137 // Branches that are isomorphic are not considered for crossover
138 DGraph test1 = gA.clone();
139 DGraph test2 = gB.clone();
140 try
141 {
142 DGraph subGraph1 = test1.extractSubgraph(gA.indexOf(vA));
143 DGraph subGraph2 = test2.extractSubgraph(gB.indexOf(vB));
144 if (maxSizeXoverSubGraph >= Math.max(subGraph1.getVertexCount(),
145 subGraph2.getVertexCount()))
146 {
147 if (!subGraph1.isIsomorphicTo(subGraph2))
148 {
149 List<Vertex> branchOnVA = new ArrayList<Vertex>();
150 branchOnVA.add(vA);
151 branchOnVA.addAll(descendantsA);
152 List<Vertex> branchOnVB = new ArrayList<Vertex>();
153 branchOnVB.add(vB);
154 branchOnVB.addAll(descendantsB);
155
156 checkAndAddXoverSites(fragSpace, branchOnVA, branchOnVB,
157 CrossoverType.BRANCH, sites);
158 }
159 }
160 } catch (DENOPTIMException e)
161 {
162 //We should never end up here.
163 e.printStackTrace();
164 }
165
166 // To limit the number of combination, we first get rid of end-point
167 // candidates that cannot be used
168 List<Vertex[]> usablePairs = new ArrayList<Vertex[]>();
169
170 // To identify subgraphs smaller than the full branch we need to find
171 // where such subgraphs end, i.e., the vertexes at the end of
172 // such subgraphs (included in them), a.k.a. the subgraph ends.
173 // Since these subgraph ends will need to allow connection with the
174 // rest of the original graph, they are related to the
175 // already-identified crossover-compatible
176 // sites, i.e., they are the parents of the vertexes collected in
177 // compatibleVrtxPairs. Yet, we can combine them
178 // * in any number from 1 to all of them,
179 // * in any combination of the chosen number of them.
180 // Also, note that the ends need not to cover all the branches. So,
181 // some combinations will have to cut some branches short while
182 // taking some other branches completely till their last leaf.
183 for (Vertex[] otherPair : compatibleVrtxPairs)
184 {
185 // NB: the xover compatible sites are the child vertexes of the
186 // subgraph ends. So we need to get the parent
187 Vertex nextToEndOnA = otherPair[0];
188 Vertex nextToEndOnB = otherPair[1];
189 Vertex endOnA = nextToEndOnA.getParent();
190 Vertex endOnB = nextToEndOnB.getParent();
191 if (endOnA==null || endOnB==null)
192 continue;
193
194 // Exclude vertexes that are not downstream to the seed of the subgraph
195 if (!descendantsA.contains(endOnA) && endOnA!=vA
196 || !descendantsB.contains(endOnB) && endOnB!=vB)
197 continue;
198
199 // If any partner is a fixed-structure templates...
200 if ((gA.getTemplateJacket()!=null
201 && gA.getTemplateJacket().getContractLevel()
202 == ContractLevel.FIXED_STRUCT)
203 || (gB.getTemplateJacket()!=null
204 && gB.getTemplateJacket().getContractLevel()
205 == ContractLevel.FIXED_STRUCT))
206 {
207 //...the two paths must have same length. This would be
208 // checked anyway later when checking for isostructural
209 // subgraphs, but here it helps reducing the size of the
210 // combinatorial problem.
211 PathSubGraph pathA = new PathSubGraph(vA, endOnA, gA);
212 PathSubGraph pathB = new PathSubGraph(vB, endOnB, gB);
213 if (pathA.getPathLength()!=pathB.getPathLength())
214 continue;
215 }
216 Vertex[] pairOfEnds = new Vertex[]{endOnA,endOnB};
217 if (!usablePairs.contains(pairOfEnds))
218 usablePairs.add(pairOfEnds);
219 }
220
221 // We classify the pairs by branch ownership
222 TreeMap<String,List<Vertex[]>> sitesByBranchIdA =
223 new TreeMap<String,List<Vertex[]>>();
224 TreeMap<String,List<Vertex[]>> sitesByBranchIdB =
225 new TreeMap<String,List<Vertex[]>>();
226 for (Vertex[] pp : usablePairs)
227 {
228 String branchIdA = gA.getBranchIdOfVertexAsStr(pp[0]);
229 String branchIdB = gB.getBranchIdOfVertexAsStr(pp[1]);
230 if (sitesByBranchIdA.containsKey(branchIdA))
231 {
232 sitesByBranchIdA.get(branchIdA).add(pp);
233 } else {
234 ArrayList<Vertex[]> lst = new ArrayList<Vertex[]>();
235 lst.add(pp);
236 sitesByBranchIdA.put(branchIdA, lst);
237 }
238 if (sitesByBranchIdB.containsKey(branchIdB))
239 {
240 sitesByBranchIdB.get(branchIdB).add(pp);
241 } else {
242 ArrayList<Vertex[]> lst = new ArrayList<Vertex[]>();
243 lst.add(pp);
244 sitesByBranchIdB.put(branchIdB, lst);
245 }
246 }
247
248 // The side with the smallest set of branches determines the max
249 // number of pairs that can define a subgraph.
250 TreeMap<String,List<Vertex[]>> fewestBranchesSide = null;
251 if (sitesByBranchIdA.size() <= sitesByBranchIdB.size())
252 fewestBranchesSide = sitesByBranchIdA;
253 else
254 fewestBranchesSide = sitesByBranchIdB;
255
256 // Add the empty, i.e., the branch with empty is not cut short.
257 for (List<Vertex[]> val : fewestBranchesSide.values())
258 val.add(new Vertex[]{null,null});
259
260 // Generate the combinations: Cartesian product of multiple lists.
261 // NB: this combinatorial generator retains the
262 // sequence of generated subsets (important for reproducibility)
263 List<List<Vertex[]>> preCombsOfEnds = Generator.cartesianProduct(
264 fewestBranchesSide.values())
265 .stream()
266 .limit(100000) //Prevent explosion!!!
267 .collect(Collectors.<List<Vertex[]>>toList());
268
269 // Remove the 'null,null' place holders that indicate the use of no
270 // end-point on a specific branch.
271 List<List<Vertex[]>> combsOfEnds = new ArrayList<List<Vertex[]>>();
272 for (List<Vertex[]> comb : preCombsOfEnds)
273 {
274 List<Vertex[]> nullPurgedComb = new ArrayList<Vertex[]>();
275 for (Vertex[] inPair : comb)
276 {
277 if (inPair[0]!=null && inPair[1]!=null)
278 nullPurgedComb.add(inPair);
279 }
280 // the case where all entries are null corresponds to use no
281 // end point on any branch, which is already considered above.
282 if (nullPurgedComb.size()>0)
283 combsOfEnds.add(nullPurgedComb);
284 }
285
286 // This would be a strategy to parallelize. However, this type of
287 // parallelization is not compatible with ensuring reproducibility.
288 // Perhaps, we could guarantee reproducibility by acting on the
289 // collector of sites as to ensure the list is sorted
290 // at the end of the generation of all possibilities.
291 /*
292 combsOfEnds
293 .parallelStream()
294 .limit(50) // Prevent explosion!
295 .forEach(c -> processCombinationOfEndPoints(pair, c, sites,
296 fragSpace));
297 */
298
299 combsOfEnds.stream()
300 .limit(50) // Prevent explosion!
301 .forEach(c -> processCombinationOfEndPoints(pair, c, sites,
302 fragSpace));
303 }
304
305 // NB: we consider only templates that are at the same level of embedding
306 for (Vertex vA : graphA.getVertexList())
307 {
308 if (!(vA instanceof Template))
309 continue;
310 Template tA = (Template) vA;
311
312 if (tA.getContractLevel() == ContractLevel.FIXED)
313 continue;
314
315 for (Vertex vB : graphB.getVertexList())
316 {
317 if (!(vB instanceof Template))
318 continue;
319 Template tB = (Template) vB;
320
321 if (tB.getContractLevel() == ContractLevel.FIXED)
322 continue;
323
324 for (XoverSite xos : locateCompatibleXOverPoints(
325 tA.getInnerGraph(), tB.getInnerGraph(), fragSpace,
326 maxSizeXoverSubGraph))
327 {
328 if (!sites.contains(xos))
329 sites.add(xos);
330 }
331 }
332 }
333 return sites;
334 }

350 private static void processCombinationOfEndPoints(Vertex[] pair,
351 List<Vertex[]> cominationOfEnds,
352 List<XoverSite> collector, FragmentSpace fragSpace)
353 {
354 // Empty set corresponds to using the entire branch and subgraph and
355 // has been already dealt with at this point
356 if (cominationOfEnds.size()==0)
357 return;
358
359 // This would be a strategy to parallelize. However, this type of
360 // parallelization is not compatible with ensuring reproducibility.
361 // Perhaps, we could guarantee reproducibility by acting on the
362 // collector as to ensure the list of collected results is sorted
363 // at the end of the generation of all possibilities
364 /*
365 List<List<Vertex[]>> permutsOfEnds = new ArrayList<List<Vertex[]>>();
366 Generator.subset(cominationOfEnds)
367 .simple()
368 .stream()
369 .limit(100) // Prevent explosion!
370 .forEach(c -> permutsOfEnds.add(c));
371 permutsOfEnds
372 .parallelStream()
373 .forEach(c -> processPermutationOfEndPoints(pair, c, collector,
374 fragSpace));
375 */
376
377 Generator.permutation(cominationOfEnds)
378 .simple()
379 .stream()
380 .limit(100) // Prevent explosion!
381 .forEach(c -> processPermutationOfEndPoints(pair, c, collector,
382 fragSpace));
383 }

399 private static void processPermutationOfEndPoints(Vertex[] pair,
400 List<Vertex[]> chosenSequenceOfEndpoints,
401 List<XoverSite> collector, FragmentSpace fragSpace)
402 {
403 Vertex vA = pair[0];
404 Vertex vB = pair[1];
405 DGraph gA = vA.getGraphOwner();
406 DGraph gB = vB.getGraphOwner();
407
408 // Exclude overlapping combinations
409 boolean exclude = false;
410 for (Vertex[] pairA : chosenSequenceOfEndpoints)
411 {
412 for (Vertex[] pairB : chosenSequenceOfEndpoints)
413 {
414 if (pairA==pairB)
415 continue;
416
417 if (pairA[0]==pairB[0] || pairA[1]==pairB[1])
418 {
419 exclude = true;
420 break;
421 }
422 }
423 if (exclude)
424 break;
425 }
426 if (exclude)
427 return;
428
429 List<Vertex> subGraphEndInA = new ArrayList<Vertex>();
430 List<Vertex> subGraphEndInB = new ArrayList<Vertex>();
431 List<Vertex> alreadyIncludedFromA = new ArrayList<Vertex>();
432 List<Vertex> alreadyIncludedFromB = new ArrayList<Vertex>();
433 for (Vertex[] otherPair : chosenSequenceOfEndpoints)
434 {
435 Vertex endOnA = otherPair[0];
436 Vertex endOnB = otherPair[1];
437
438 // Ignore vertexes that are already part of the subgraph
439 if (alreadyIncludedFromA.contains(endOnA)
440 || alreadyIncludedFromB.contains(endOnB))
441 continue;
442
443 PathSubGraph pathA = new PathSubGraph(vA, endOnA, gA);
444 PathSubGraph pathB = new PathSubGraph(vB, endOnB, gB);
445 subGraphEndInA.add(endOnA);
446 subGraphEndInB.add(endOnB);
447 alreadyIncludedFromA.addAll(pathA.getVertecesPath());
448 alreadyIncludedFromB.addAll(pathB.getVertecesPath());
449 }
450 ArrayList<Vertex> subGraphA = new ArrayList<Vertex>();
451 subGraphA.add(vA);
452 if (!subGraphEndInA.contains(vA))
453 gA.getChildTreeLimited(vA, subGraphA, subGraphEndInA, true);
454
455 ArrayList<Vertex> subGraphB = new ArrayList<Vertex>();
456 subGraphB.add(vB);
457 if (!subGraphEndInB.contains(vB))
458 gB.getChildTreeLimited(vB, subGraphB, subGraphEndInB, true);
459
460 // The two subgraphs must not be isomorfic to prevent unproductive crossover
461 if (subGraphA.size()>1 && subGraphB.size()>1)
462 {
463 DGraph subGraphCloneA = gA.extractSubgraph(subGraphA);
464 DGraph subGraphCloneB = gB.extractSubgraph(subGraphB);
465 if (subGraphCloneA.isIsomorphicTo(subGraphCloneB))
466 return;
467 } else {
468 if (subGraphA.get(0).sameAs(subGraphB.get(0), new StringBuilder()))
469 return;
470 }
471
472 checkAndAddXoverSites(fragSpace, subGraphA, subGraphB,
473 CrossoverType.SUBGRAPH, collector);
474 }

487 private static void checkAndAddXoverSites(FragmentSpace fragSpace,
488 List<Vertex> subGraphA,
489 List<Vertex> subGraphB, CrossoverType xoverType,
490 List<XoverSite> collector)
491 {
492 DGraph gOwnerA = subGraphA.get(0).getGraphOwner();
493 DGraph gOwnerB = subGraphB.get(0).getGraphOwner();
494
495 // What APs need to find a corresponding AP in the other
496 // subgraph in order to allow swapping?
497 List<AttachmentPoint> needyAPsA = gOwnerA.getInterfaceAPs(
498 subGraphA);
499 List<AttachmentPoint> allAPsA = gOwnerA.getSubgraphAPs(
500 subGraphA);
501 List<AttachmentPoint> needyAPsB = gOwnerB.getInterfaceAPs(
502 subGraphB);
503 List<AttachmentPoint> allAPsB = gOwnerB.getSubgraphAPs(
504 subGraphB);
505 if (allAPsA.size() < needyAPsB.size()
506 || allAPsB.size() < needyAPsA.size())
507 {
508 // Impossible to satisfy needy APs. Crossover site is not usable.
509 return;
510 }
511
512 // Shortcut: since the compatibility of one AP that is needed to do
513 // branch swapping is guaranteed by the identification of the seeds of
514 // swappable subgraphs, we can avoid searching for a valid A mapping
515 // when the graphs are not embedded. This because any AP that is not
516 // the one connecting the subgraph to the parent can be left free or
517 // removed without side effects on templates that embed the graph
518 // because there is no such template.
519 Template jacketTmplA = gOwnerA.getTemplateJacket();
520 Template jacketTmplB = gOwnerB.getTemplateJacket();
521 if (xoverType == CrossoverType.BRANCH
522 && jacketTmplA==null
523 && jacketTmplB==null)
524 {
525 XoverSite xos = new XoverSite(subGraphA, needyAPsA, subGraphB,
526 needyAPsB, xoverType);
527 if (!collector.contains(xos))
528 collector.add(xos);
529 return;
530 }
531
532 if ((jacketTmplA!=null && jacketTmplA.getContractLevel()
533 ==ContractLevel.FIXED_STRUCT)
534 || (jacketTmplB!=null && jacketTmplB.getContractLevel()
535 ==ContractLevel.FIXED_STRUCT))
536 {
537 // Avoid to alter cyclicity of inner graphs
538 for (Vertex v : subGraphA)
539 if (v.isRCV() && !gOwnerA.getRingsInvolvingVertex(v).isEmpty())
540 return;
541 for (Vertex v : subGraphB)
542 if (v.isRCV() && !gOwnerB.getRingsInvolvingVertex(v).isEmpty())
543 return;
544
545 // Avoid to change the structure of inner graphs
546 DGraph subGraphCloneA = gOwnerA.extractSubgraph(subGraphA);
547 DGraph subGraphCloneB = gOwnerB.extractSubgraph(subGraphB);
548 if (!subGraphCloneA.isIsostructuralTo(subGraphCloneB))
549 return;
550 }
551
552 // Retain connection to parent to keep directionality of spanning tree!
553 // To this end, identify the seed of the subgraphs...
554 Vertex seedOnA = gOwnerA.getDeepestAmongThese(subGraphA);
555 Vertex seedOnB = gOwnerB.getDeepestAmongThese(subGraphB);
556 //...and ensure we use the same APs to link to the parent graph.
557 APMapping fixedRootAPs = new APMapping();
558 fixedRootAPs.put(seedOnA.getEdgeToParent().getTrgAP(),
559 seedOnB.getEdgeToParent().getTrgAP());
560
561 APMapFinder apmf = new APMapFinder(fragSpace,
562 allAPsA, needyAPsA, allAPsB, needyAPsB,
563 fixedRootAPs,
564 false, // false: we stop at the first good mapping
565 true, // true: only complete mapping
566 false); // false: free APs are not compatible by default
567 if (apmf.foundMapping())
568 {
569 XoverSite xos = new XoverSite(subGraphA, needyAPsA,
570 subGraphB, needyAPsB, xoverType);
571 if (!collector.contains(xos))
572 collector.add(xos);
573 }
574 }

589 private static boolean isCrossoverPossible(Edge eA, Edge eB,
590 FragmentSpace fragSpace)
591 {
592 APClass apClassSrcA = eA.getSrcAPClass();
593 APClass apClassTrgA = eA.getTrgAPClass();
594 APClass apClassSrcB = eB.getSrcAPClass();
595 APClass apClassTrgB = eB.getTrgAPClass();
596
597 if (apClassSrcA.isCPMapCompatibleWith(apClassTrgB, fragSpace))
598 {
599 if (apClassSrcB.isCPMapCompatibleWith(apClassTrgA, fragSpace))
600 {
601 return true;
602 }
603 }
604 return false;
605 }

749 + "vertex (AP "+chosenAPId+" of vertex "+vertex+").");
750 }
751 return extendLink(e, chosenNewVrtxId, mnt, fragSpace);
752 }

768 public static boolean extendLink(Edge edge, int chosenBBIdx,
769 Monitor mnt, FragmentSpace fragSpace) throws DENOPTIMException
770 {
771 //TODO: for reproducibility, the AP mapping should become an optional
772 // parameter: if given we try to use it, if not given we GraphLinkFinder

775 GraphLinkFinder glf = null;
776 if (chosenBBIdx < 0)

783 {
784 mnt.increase(CounterID.FAILEDMUTATTEMTS_PERFORM_NOADDLINK_FIND);

786 }
787
788 // Need to convert the mapping to make it independent from the instance

791 new LinkedHashMap<AttachmentPoint,Integer>();
792 for (Entry<AttachmentPoint, AttachmentPoint> e :

795 apMap.put(e.getKey(), e.getValue().getIndexInOwner());
796 }
797
798 DGraph graph = edge.getSrcAP().getOwner().getGraphOwner();
799 boolean done = graph.insertVertex(edge,
800 glf.getChosenAlternativeLink().getBuildingBlockId(),
801 glf.getChosenAlternativeLink().getBuildingBlockType(),
802 apMap, fragSpace);
803 if (!done)
804 {
805 mnt.increase(
806 CounterID.FAILEDMUTATTEMTS_PERFORM_NOADDLINK_EDIT);
807 }
808 return done;

834 protected static boolean rebuildBranch(Vertex vertex,
835 boolean force, int chosenVrtxIdx, int chosenApId,
836 GAParameters settings) throws DENOPTIMException
837 {
838 DGraph g = vertex.getGraphOwner();
839
840 // first get the edge with the parent
841 Edge e = vertex.getEdgeToParent();
842 if (e == null)
843 {
844 String msg = "Program Bug in substituteFragment: Unable to locate "
845 + "parent edge for vertex "+vertex+" in graph "+g;
846 settings.getLogger().log(Level.SEVERE, msg);
847 throw new DENOPTIMException(msg);
848 }
849
850 // vertex id of the parent
851 long pvid = e.getSrcVertex();
852 Vertex parentVrt = g.getVertexWithId(pvid);
853
854 // Need to remember symmetry because we are deleting the symm. vertices
855 boolean symmetry = g.hasSymmetryInvolvingVertex(vertex);
856
857 // delete the vertex and its children and all its symmetric partners
858 deleteFragment(vertex);
859
860 // extend the graph at this vertex but without recursion
861 return extendGraph(parentVrt,false,symmetry,force,chosenVrtxIdx,
862 chosenApId, settings);
863 }

994 protected static boolean extendGraph(Vertex curVrtx,
995 boolean extend,
996 boolean symmetryOnAps,
997 boolean force,
998 int chosenVrtxIdx,
999 int chosenApId,
1000 GAParameters settings)
1001 throws DENOPTIMException
1002 {
1003 // get settings //TODO: this should happen inside RunTimeParameters
1004 RingClosureParameters rcParams = new RingClosureParameters();
1005 if (settings.containsParameters(ParametersType.RC_PARAMS))
1006 {
1007 rcParams = (RingClosureParameters)settings.getParameters(
1008 ParametersType.RC_PARAMS);
1009 }
1010 FragmentSpaceParameters fsParams = new FragmentSpaceParameters();
1011 if (settings.containsParameters(ParametersType.FS_PARAMS))
1012 {
1013 fsParams = (FragmentSpaceParameters)settings.getParameters(
1014 ParametersType.FS_PARAMS);
1015 }
1016 int maxHeavyAtoms = fsParams.getMaxHeavyAtom();
1017
1018 // return true if the append has been successful
1019 boolean status = false;
1020
1021 // check if the fragment has available APs
1022 if (!curVrtx.hasFreeAP())
1023 {
1024 return status;
1025 }
1026
1027 DGraph molGraph = curVrtx.getGraphOwner();
1028 int lvl = molGraph.getLevel(curVrtx);
1029
1030 ArrayList<Long> addedVertices = new ArrayList<>();
1031
1032 List<AttachmentPoint> lstDaps = curVrtx.getAttachmentPoints();
1033 List<AttachmentPoint> toDoAPs = new ArrayList<AttachmentPoint>();
1034 toDoAPs.addAll(lstDaps);
1035 for (int i=0; i<lstDaps.size(); i++)
1036 {
1037 // WARNING: randomization decouples 'i' from the index of the AP
1038 // in the vertex's list of APs! So 'i' is just the i-th attempt on
1039 // the curVertex.
1040
1041 AttachmentPoint ap = settings.getRandomizer().randomlyChooseOne(
1042 toDoAPs);
1043 toDoAPs.remove(ap);
1044 int apId = ap.getIndexInOwner();
1045
1046 // is it possible to extend on this AP?
1047 if (!ap.isAvailable())
1048 {
1049 continue;
1050 }
1051
1052 // Ring closure does not change the size of the molecule, so we
1053 // give it an extra chance to occur irrespectively on molecular size
1054 // limit, while still subject of crowdedness probability.
1055 boolean allowOnlyRingClosure = false;
1056 if (!force)
1057 {
1058 // Decide whether we want to extend the graph at this AP?
1059 // Note that depending on the criterion (level/molSize) one
1060 // of these two first factors is 1.0.
1061 double molSizeProb = EAUtils.getMolSizeProbability(molGraph,
1062 settings);
1063 double byLevelProb = EAUtils.getGrowthByLevelProbability(lvl,
1064 settings);
1065 double crowdingProb = EAUtils.getCrowdingProbability(ap,
1066 settings);
1067 double extendGraphProb = molSizeProb * byLevelProb * crowdingProb;
1068 boolean fgrow = settings.getRandomizer().nextBoolean(
1069 extendGraphProb);
1070 if (!fgrow)
1071 {
1072 if (rcParams.allowRingClosures()
1073 && settings.getRandomizer().nextBoolean(byLevelProb
1074 * crowdingProb))
1075 {
1076 allowOnlyRingClosure = true;
1077 } else {
1078 continue;
1079 }
1080 }
1081 }
1082
1083 // Apply closability bias in selection of next fragment
1084 if (!allowOnlyRingClosure
1085 && rcParams.allowRingClosures()
1086 && rcParams.selectFragmentsFromClosableChains())
1087 {
1088 boolean successful = attachFragmentInClosableChain(curVrtx,
1089 apId, molGraph, addedVertices, settings);
1090 if (successful)
1091 {
1092 continue;
1093 }
1094 }
1095
1096 // find a compatible combination of fragment and AP
1097 IdFragmentAndAP chosenFrgAndAp = null;
1098 if (allowOnlyRingClosure)
1099 {
1100 chosenFrgAndAp = getRCVForSrcAp(curVrtx, apId,
1101 fsParams.getFragmentSpace());
1102 } else {
1103 chosenFrgAndAp = getFrgApForSrcAp(curVrtx,
1104 apId, chosenVrtxIdx, chosenApId, fsParams.getFragmentSpace());
1105 }
1106 int fid = chosenFrgAndAp.getVertexMolId();
1107 if (fid == -1)
1108 {
1109 continue;
1110 }
1111
1112 // Stop if graph is already too big
1113 Vertex incomingVertex = Vertex.newVertexFromLibrary(-1,
1114 chosenFrgAndAp.getVertexMolId(),
1115 BBType.FRAGMENT,
1116 fsParams.getFragmentSpace());
1117 if ((curVrtx.getGraphOwner().getHeavyAtomsCount() +
1118 incomingVertex.getHeavyAtomsCount()) > maxHeavyAtoms)
1119 {
1120 continue;
1121 }
1122
1123 // Decide on symmetric substitution within this vertex...
1124 boolean cpOnSymAPs = applySymmetry(
1125 fsParams.getFragmentSpace().imposeSymmetryOnAPsOfClass(
1126 ap.getAPClass()),
1127 settings.getSymmetryProbability(),
1128 fsParams.getRandomizer());
1129 SymmetricAPs symAPs = new SymmetricAPs();
1130 if (curVrtx.getSymmetricAPs(ap).size()!=0
1131 && (cpOnSymAPs || symmetryOnAps)
1132 && !allowOnlyRingClosure)
1133 {
1134 symAPs.addAll(curVrtx.getSymmetricAPs(ap));
1135
1136 // Are symmetric APs rooted on same atom?
1137 boolean allOnSameSrc = true;
1138 for (AttachmentPoint symAP : symAPs)
1139 {
1140 if (!symAP.hasSameSrcAtom(ap))
1141 {
1142 allOnSameSrc = false;
1143 break;
1144 }
1145 }
1146
1147 if (allOnSameSrc)
1148 {
1149 // If the APs are rooted on the same src atom, we want to
1150 // apply the crowdedness probability to avoid over crowded
1151 // atoms
1152
1153 int crowdedness = EAUtils.getCrowdedness(ap);
1154
1155 SymmetricAPs toKeep = new SymmetricAPs();
1156
1157 // Start by keeping "ap"
1158 toKeep.add(ap);
1159 crowdedness = crowdedness + 1;
1160
1161 // Pick the accepted value once (used to decide how much
1162 // crowdedness we accept)
1163 double shot = settings.getRandomizer().nextDouble();
1164
1165 // Keep as many as allowed by the crowdedness decision
1166 for (AttachmentPoint symAP : symAPs)
1167 {
1168 if (symAP == ap)
1169 continue;
1170
1171 double crowdProb = EAUtils.getCrowdingProbability(
1172 crowdedness, settings);
1173
1174 if (shot > crowdProb)
1175 break;
1176
1177 toKeep.add(symAP);
1178 crowdedness = crowdedness + 1;
1179 }
1180
1181 // Adjust the list of symmetric APs to work with
1182 symAPs = toKeep;
1183 }
1184 } else {
1185 symAPs = new SymmetricAPs();
1186 symAPs.add(ap);
1187 }
1188
1189 // ...and inherit symmetry from previous levels
1190 boolean cpOnSymVrts = molGraph.hasSymmetryInvolvingVertex(curVrtx);
1191 SymmetricVertexes symVerts = new SymmetricVertexes();
1192 if (cpOnSymVrts)
1193 {
1194 symVerts = molGraph.getSymSetForVertex(curVrtx);
1195 }
1196 else
1197 {
1198 symVerts.add(curVrtx);
1199 }
1200
1201 // Consider size after application of symmetry
1202 if ((curVrtx.getGraphOwner().getHeavyAtomsCount() +
1203 incomingVertex.getHeavyAtomsCount() * symVerts.size()
1204 * symAPs.size()) > maxHeavyAtoms)
1205 {
1206 continue;
1207 }
1208
1209 GraphUtils.ensureVertexIDConsistency(molGraph.getMaxVertexId());
1210
1211 // loop on all symmetric vertices, but can be only one.
1212 SymmetricVertexes newSymSetOfVertices = new SymmetricVertexes();
1213 for (AttachmentPoint symAP : symAPs)
1214 {
1215 if (!symAP.isAvailable())
1216 {
1217 continue;
1218 }
1219
1220 // Finally add the fragment on a symmetric AP
1221 long newVrtId = GraphUtils.getUniqueVertexIndex();
1222 Vertex fragVertex = Vertex.newVertexFromLibrary(newVrtId,
1223 chosenFrgAndAp.getVertexMolId(),
1224 BBType.FRAGMENT,
1225 fsParams.getFragmentSpace());
1226 AttachmentPoint trgAP = fragVertex.getAP(
1227 chosenFrgAndAp.getApId());
1228
1229 molGraph.appendVertexOnAP(symAP, trgAP);
1230
1231 addedVertices.add(newVrtId);
1232 newSymSetOfVertices.add(fragVertex);
1233 }
1234
1235 // If any, store symmetry of new vertices in the graph
1236 if (newSymSetOfVertices.size() > 1)
1237 {
1238 molGraph.addSymmetricSetOfVertices(newSymSetOfVertices);
1239 }
1240 } // end loop over APs
1241
1242 if (extend)
1243 {
1244 // attempt to further extend each of the newly added vertices
1245 for (int i=0; i<addedVertices.size(); i++)
1246 {
1247 long vid = addedVertices.get(i);
1248 Vertex v = molGraph.getVertexWithId(vid);
1249 extendGraph(v, extend, symmetryOnAps, settings);
1250 }
1251 }
1252
1253 if (addedVertices.size() > 0)
1254 status = true;
1255
1256 return status;
1257 }

1513
1519 protected static ArrayList<FragForClosabChains> getFragmentForClosableChain(
1520 Vertex curVertex,
1521 int dapidx,
1522 DGraph molGraph)
1523 throws DENOPTIMException
1524 {
1525 // Select candidate fragments respecting the closability conditions
1526 ArrayList<FragForClosabChains> lstChosenFfCc =
1527 new ArrayList<FragForClosabChains>();
1528
1529 if (molGraph.getClosableChains().size() == 0)
1530 {
1531 return lstChosenFfCc;
1532 }
1533
1534 if (curVertex.getBuildingBlockType() == BBType.SCAFFOLD)
1535 {
1536 for (ClosableChain cc : molGraph.getClosableChains())
1537 {
1538 int posInCc = cc.involvesVertex(curVertex);
1539 ChainLink cLink = cc.getLink(posInCc);
1540 int nfid = -1;
1541 BBType nfty = BBType.UNDEFINED;
1542 int nfap = -1;
1543
1544 if (cLink.getApIdToLeft() != dapidx &&
1545 cLink.getApIdToRight() != dapidx)
1546 {
1547 // Chain does not involve AP dapidx
1548 continue;
1549 }
1550
1551 if (cLink.getApIdToRight() == dapidx)
1552 {
1553 if (cc.getSize() > (posInCc+1))
1554 {
1555 // cLink is NOT the rightmost chain link
1556 ChainLink nextChainLink = cc.getLink(posInCc+1);
1557 nfid = nextChainLink.getIdx();
1558 nfty = nextChainLink.getFragType();
1559 nfap = nextChainLink.getApIdToLeft();
1560 }
1561 else
1562 {
1563 // cLink is the rightmost chain link
1564 // closability bias suggest NO fragment
1565 }
1566 }
1567 else if (cLink.getApIdToLeft() == dapidx)
1568 {
1569 if ((posInCc-1) >= 0)
1570 {
1571 // cLink is NOT the leftmost chain link
1572 ChainLink nextChainLink = cc.getLink(posInCc-1);
1573 nfid = nextChainLink.getIdx();
1574 nfty = nextChainLink.getFragType();
1575 nfap = nextChainLink.getApIdToRight();
1576 }
1577 else
1578 {
1579 // cLink is the leftmost chain link
1580 // closability bias suggest NO fragment
1581 }
1582 }
1583
1584 ArrayList<Integer> eligibleFrgId = new ArrayList<Integer>();
1585 eligibleFrgId.add(nfid);
1586 eligibleFrgId.add(nfty.toOldInt());
1587 eligibleFrgId.add(nfap);
1588 boolean found = false;
1589 for (FragForClosabChains ffcc : lstChosenFfCc)
1590 {
1591 int fidA = ffcc.getFragIDs().get(0);
1592 BBType ftyA = BBType.parseInt(ffcc.getFragIDs().get(1));
1593 int fapA = ffcc.getFragIDs().get(2);
1594 if (nfid==fidA && nfty==ftyA && nfap==fapA)
1595 {
1596 found = true;
1597 ffcc.getCompatibleCC().add(cc);
1598 }
1599 else
1600 {
1601 ffcc.getIncompatibleCC().add(cc);
1602 }
1603 }
1604 if (!found)
1605 {
1606 ArrayList<ClosableChain> compatChains =
1607 new ArrayList<ClosableChain>();
1608 ArrayList<ClosableChain> incompatChains =
1609 new ArrayList<ClosableChain>();
1610 for (FragForClosabChains otherFfCc : lstChosenFfCc)
1611 {
1612 incompatChains.addAll(otherFfCc.getCompatibleCC());
1613 }
1614 FragForClosabChains newChosenCc = new FragForClosabChains(
1615 compatChains,
1616 incompatChains,
1617 eligibleFrgId);
1618
1619 newChosenCc.addCompatibleCC(cc);
1620 lstChosenFfCc.add(newChosenCc);
1621 }
1622 }
1623 }
1624 else
1625 {
1626 Vertex parent = molGraph.getParent(curVertex);
1627 Edge edge = molGraph.getEdgeWithParent(
1628 curVertex.getVertexId());
1629 int prntId = parent.getBuildingBlockId();
1630 BBType prntTyp = parent.getBuildingBlockType();
1631 int prntAp = edge.getSrcAPID();
1632 int chidAp = edge.getTrgAPID();
1633 for (ClosableChain cc : molGraph.getClosableChains())
1634 {
1635 int posInCc = cc.involvesVertexAndAP(curVertex, dapidx, chidAp);
1636
1637 if (posInCc == -1)
1638 {
1639 // closable chain does not span this combination
1640 // of vertex and APs
1641 continue;
1642 }
1643
1644 ChainLink cLink = cc.getLink(posInCc);
1645 int nfid = -1;
1646 BBType nfty = BBType.UNDEFINED;
1647 int nfap = -1;
1648
1649 List<Integer> altertnativeDirections = new ArrayList<>();
1650 altertnativeDirections.add(-1);
1651 altertnativeDirections.add(+1);
1652 for (int altDir : altertnativeDirections)
1653 {
1654 ChainLink parentLink = cc.getLink(posInCc + altDir);
1655 int pLnkId = parentLink.getIdx();
1656 BBType pLnkTyp = parentLink.getFragType();
1657
1658 int pLnkAp = -1;
1659 int cApId = -1;
1660 if (altDir>0)
1661 {
1662 pLnkAp = parentLink.getApIdToRight();
1663 cApId = cLink.getApIdToLeft();
1664 } else {
1665 pLnkAp = parentLink.getApIdToLeft();
1666 cApId = cLink.getApIdToRight();
1667 }
1668 if (pLnkId==prntId && pLnkTyp==prntTyp && pLnkAp == prntAp &&
1669 cApId == chidAp)
1670 {
1671 if (cc.getSize() > (posInCc+altDir)
1672 && (posInCc+altDir) >= 0)
1673 {
1674 ChainLink nextChainLink = cc.getLink(posInCc + altDir);
1675 nfid = nextChainLink.getIdx();
1676 nfty = nextChainLink.getFragType();
1677 nfap = nextChainLink.getApIdToLeft();

1746 public static boolean performCrossover(XoverSite site,
1747 FragmentSpace fragSpace) throws DENOPTIMException
1748 {
1749 DGraph gA = site.getA().get(0).getGraphOwner();
1750 DGraph gB = site.getB().get(0).getGraphOwner();
1751
1752 // All the APs that point away from the subgraph
1753 List<AttachmentPoint> allAPsOnA = gA.getSubgraphAPs(site.getA());
1754 List<AttachmentPoint> allAPsOnB = gB.getSubgraphAPs(site.getB());
1755
1756 // The APs that are required to have a mapping for proper crossover,
1757 // eg. because the change of subgraph needs to retain a super structure.
1758 List<AttachmentPoint> needyAPsOnA = site.getAPsNeedingMappingA();
1759 List<AttachmentPoint> needyAPsOnB = site.getAPsNeedingMappingB();
1760
1761 // APs that connects the subgraphs' root to the parent vertex
1762 AttachmentPoint apToParentA = null;
1763 AttachmentPoint apToParentB = null;
1764 for (AttachmentPoint ap : needyAPsOnA)
1765 {
1766 if (!ap.isSrcInUserThroughout())
1767 {
1768 apToParentA = ap;
1769 //WARNING: we assume there is one AND only one AP to parent!!!
1770 break;
1771 }
1772 }
1773 for (AttachmentPoint ap : needyAPsOnB)
1774 {
1775 if (!ap.isSrcInUserThroughout())
1776 {
1777 apToParentB = ap;
1778 //WARNING: we assume there is one AND only one AP to parent!!!
1779 break;
1780 }
1781 }
1782 if (apToParentA==null || apToParentB==null)
1783 {
1784 throw new DENOPTIMException("Could not identify any attachment "
1785 + "point connecting a subgraph to the rest of the graph. "
1786 + "This violates assumption that crossover does not "
1787 + "involve scaffold or vertexes without parent.");
1788 }
1789 // Check if the subgraphs can be used with reversed edge direction, or
1790 // bias the AP mapping to use the original source vertexes.
1791 APMapping fixedRootAPs = null;
1792 //TODO: REACTIVATE ONCE REVERSION of the subgrsph's spanning tree is in place.
1793 //if (!subG_M.isReversible() || !subG_F.isReversible())
1794 if (true)
1795 {
1796 // Constrain the AP mapping so that the AP originally used to
1797 // connect to the parent vertex, will also be used the same way.
1798 // Thus, forces retention of edge direction within the subgraph.
1799 fixedRootAPs = new APMapping();
1800 fixedRootAPs.put(apToParentA, apToParentB);
1801 }
1802
1803 // Find an AP mapping that satisfies both root-vertex constrain and
1804 // need to ensure the mapping of needy APs
1805 APMapFinder apmf = new APMapFinder(fragSpace,
1806 allAPsOnA, needyAPsOnA,
1807 allAPsOnB, needyAPsOnB, fixedRootAPs,
1808 false, // false means stop at the first compatible mapping.
1809 false, // false means we do not require complete mapping.
1810 false); // false means free AP are not considered compatible.
1811 if (!apmf.foundMapping())
1812 {
1813 // Since the xover site has been detected by searching for compatible
1814 // sites that do have a mapping there should always be at least one
1815 // mapping and this exception should never occur.
1816 throw new DENOPTIMException("Missing AP mapping for known XoverSite.");
1817 }
1818
1819 // To replace each subgraph in the original graphs, we need to
1820 // map the APs on the original A/B graph with those in the
1821 // corresponding incoming subgraph, which are clones of the original:
1822 DGraph subGraphA = gA.extractSubgraph(site.getA());
1823 DGraph subGraphB = gB.extractSubgraph(site.getB());
1824
1825 // Here we create the two AP mappings we need: one for A other for B.
1826 LinkedHashMap<AttachmentPoint,AttachmentPoint>
1827 apMapA = new LinkedHashMap<AttachmentPoint,AttachmentPoint>();
1828 LinkedHashMap<AttachmentPoint,AttachmentPoint>
1829 apMapB = new LinkedHashMap<AttachmentPoint,AttachmentPoint>();
1830 for (Map.Entry<AttachmentPoint, AttachmentPoint> e :
1831 apmf.getChosenAPMapping().entrySet())
1832 {
1833 AttachmentPoint apOnA = e.getKey();
1834 AttachmentPoint apOnB = e.getValue();
1835
1836 // NB: assumption that vertex IDs are healthy, AND that order of APs
1837 // is retained upon cloning of the subgraph!
1838 AttachmentPoint apOnSubGraphA = subGraphA.getVertexWithId(
1839 apOnA.getOwner().getVertexId()).getAP(
1840 apOnA.getIndexInOwner());
1841 AttachmentPoint apOnSubGraphB = subGraphB.getVertexWithId(
1842 apOnB.getOwner().getVertexId()).getAP(
1843 apOnB.getIndexInOwner());
1844 apMapA.put(apOnA, apOnSubGraphB);
1845 apMapB.put(apOnB, apOnSubGraphA);
1846 }
1847
1848 // Now we do the actual replacement of subgraphs
1849 if (!gA.replaceSubGraph(site.getA(), subGraphB, apMapA, fragSpace))
1850 return false;
1851 if (!gB.replaceSubGraph(site.getB(), subGraphA, apMapB, fragSpace))
1852 return false;
1853
1854 return true;
1855 }
